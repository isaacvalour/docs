## Overview

Get information about the currently authenticated user. This endpoint returns the user's profile data, permissions, and account status.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

## Use Cases

### User Profile Display

<CodeGroup>

```javascript JavaScript
class UserProfile {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async loadUserProfile() {
    try {
      const response = await this.apiClient.get('/v1/users/me');
      
      if (response.success) {
        this.displayUserProfile(response.data);
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to load user profile:', error);
      throw error;
    }
  }
  
  displayUserProfile(userData) {
    // Update UI elements
    document.getElementById('user-name').textContent = 
      `${userData.first_name} ${userData.last_name}`;
    document.getElementById('user-email').textContent = userData.email;
    document.getElementById('user-role').textContent = userData.role;
    document.getElementById('user-status').textContent = userData.status;
    
    // Show verification status
    if (userData.email_verified) {
      document.getElementById('email-status').textContent = 'Verified';
      document.getElementById('email-status').className = 'status verified';
    } else {
      document.getElementById('email-status').textContent = 'Unverified';
      document.getElementById('email-status').className = 'status unverified';
    }
    
    // Show 2FA status
    if (userData['2fa_enabled']) {
      document.getElementById('2fa-status').textContent = 'Enabled';
      document.getElementById('2fa-status').className = 'status enabled';
    } else {
      document.getElementById('2fa-status').textContent = 'Disabled';
      document.getElementById('2fa-status').className = 'status disabled';
    }
    
    // Display quotas
    this.displayQuotas(userData.quotas);
  }
  
  displayQuotas(quotas) {
    if (quotas.virtual_accounts) {
      const vaQuota = quotas.virtual_accounts;
      document.getElementById('va-quota').textContent = 
        `${vaQuota.used}/${vaQuota.limit} (${vaQuota.remaining} remaining)`;
    }
  }
}

// Usage
const userProfile = new UserProfile(apiClient);
await userProfile.loadUserProfile();
```

</CodeGroup>

### Permission Checking

<CodeGroup>

```javascript JavaScript
class PermissionChecker {
  constructor(apiClient) {
    this.apiClient = apiClient;
    this.userData = null;
  }
  
  async loadUserData() {
    try {
      const response = await this.apiClient.get('/v1/users/me');
      if (response.success) {
        this.userData = response.data;
        return this.userData;
      }
    } catch (error) {
      console.error('Failed to load user data:', error);
    }
    return null;
  }
  
  hasPermission(permission) {
    if (!this.userData || !this.userData.permissions) {
      return false;
    }
    return this.userData.permissions.includes(permission);
  }
  
  canReadVirtualAccounts() {
    return this.hasPermission('read:virtual_accounts');
  }
  
  canWriteVirtualAccounts() {
    return this.hasPermission('write:virtual_accounts');
  }
  
  canReadTransactions() {
    return this.hasPermission('read:transactions');
  }
  
  canWriteTransactions() {
    return this.hasPermission('write:transactions');
  }
  
  isAdmin() {
    return this.userData && this.userData.role === 'admin';
  }
  
  isSuperAdmin() {
    return this.userData && this.userData.role === 'super_admin';
  }
}

// Usage
const permissionChecker = new PermissionChecker(apiClient);
await permissionChecker.loadUserData();

if (permissionChecker.canReadVirtualAccounts()) {
  // Show virtual accounts
  loadVirtualAccounts();
} else {
  // Hide virtual accounts section
  document.getElementById('virtual-accounts').style.display = 'none';
}
```

</CodeGroup>

### Quota Monitoring

<CodeGroup>

```javascript JavaScript
class QuotaMonitor {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async checkQuotas() {
    try {
      const response = await this.apiClient.get('/v1/users/me');
      
      if (response.success) {
        const quotas = response.data.quotas;
        this.displayQuotaStatus(quotas);
        return quotas;
      }
    } catch (error) {
      console.error('Failed to check quotas:', error);
    }
    return null;
  }
  
  displayQuotaStatus(quotas) {
    if (quotas.virtual_accounts) {
      const vaQuota = quotas.virtual_accounts;
      const percentage = (vaQuota.used / vaQuota.limit) * 100;
      
      // Update progress bar
      const progressBar = document.getElementById('va-quota-progress');
      progressBar.style.width = `${percentage}%`;
      
      // Update text
      document.getElementById('va-quota-text').textContent = 
        `${vaQuota.used}/${vaQuota.limit} virtual accounts`;
      
      // Show warning if approaching limit
      if (percentage > 80) {
        document.getElementById('va-quota-warning').style.display = 'block';
      }
    }
  }
  
  canCreateVirtualAccount() {
    // This would need to be called after checkQuotas()
    const vaQuota = this.userData?.quotas?.virtual_accounts;
    return vaQuota && vaQuota.remaining > 0;
  }
}

// Usage
const quotaMonitor = new QuotaMonitor(apiClient);
await quotaMonitor.checkQuotas();
```
