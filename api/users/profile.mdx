## Overview

Get detailed user profile information including personal details, preferences, and account settings. This endpoint provides comprehensive user data for profile management.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

## Use Cases

### Profile Dashboard

<CodeGroup>

```javascript JavaScript
class ProfileDashboard {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async loadProfile() {
    try {
      const response = await this.apiClient.get('/v1/users/profile');
      
      if (response.success) {
        this.displayProfile(response.data);
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to load profile:', error);
      throw error;
    }
  }
  
  displayProfile(profileData) {
    // Display basic information
    this.displayBasicInfo(profileData);
    
    // Display preferences
    this.displayPreferences(profileData.preferences);
    
    // Display statistics
    this.displayStatistics(profileData.statistics);
    
    // Display quotas
    this.displayQuotas(profileData.quotas);
  }
  
  displayBasicInfo(userData) {
    document.getElementById('user-name').textContent = 
      `${userData.first_name} ${userData.last_name}`;
    document.getElementById('user-email').textContent = userData.email;
    document.getElementById('user-role').textContent = userData.role;
    document.getElementById('user-status').textContent = userData.status;
    document.getElementById('user-country').textContent = userData.country;
    
    // Show verification status
    const emailStatus = document.getElementById('email-status');
    if (userData.email_verified) {
      emailStatus.textContent = 'Verified';
      emailStatus.className = 'status verified';
    } else {
      emailStatus.textContent = 'Unverified';
      emailStatus.className = 'status unverified';
    }
    
    // Show 2FA status
    const twoFactorStatus = document.getElementById('2fa-status');
    if (userData['2fa_enabled']) {
      twoFactorStatus.textContent = 'Enabled';
      twoFactorStatus.className = 'status enabled';
    } else {
      twoFactorStatus.textContent = 'Disabled';
      twoFactorStatus.className = 'status disabled';
    }
  }
  
  displayPreferences(preferences) {
    document.getElementById('language').textContent = preferences.language;
    document.getElementById('timezone').textContent = preferences.timezone;
    document.getElementById('currency').textContent = preferences.currency;
    
    // Display notification preferences
    const notifications = preferences.notifications;
    document.getElementById('email-notifications').checked = notifications.email;
    document.getElementById('sms-notifications').checked = notifications.sms;
    document.getElementById('push-notifications').checked = notifications.push;
    document.getElementById('webhook-notifications').checked = notifications.webhook;
  }
  
  displayStatistics(statistics) {
    document.getElementById('total-transactions').textContent = statistics.total_transactions;
    document.getElementById('total-volume').textContent = `$${statistics.total_volume.toLocaleString()}`;
    document.getElementById('virtual-accounts-created').textContent = statistics.virtual_accounts_created;
    document.getElementById('recipients-created').textContent = statistics.recipients_created;
    
    if (statistics.last_transaction) {
      const lastTransaction = new Date(statistics.last_transaction);
      document.getElementById('last-transaction').textContent = lastTransaction.toLocaleDateString();
    }
  }
  
  displayQuotas(quotas) {
    // Virtual accounts quota
    if (quotas.virtual_accounts) {
      const vaQuota = quotas.virtual_accounts;
      const vaPercentage = (vaQuota.used / vaQuota.limit) * 100;
      
      document.getElementById('va-quota-progress').style.width = `${vaPercentage}%`;
      document.getElementById('va-quota-text').textContent = 
        `${vaQuota.used}/${vaQuota.limit} (${vaQuota.remaining} remaining)`;
    }
    
    // Transactions quota
    if (quotas.transactions) {
      const txQuota = quotas.transactions;
      const txPercentage = (txQuota.daily_used / txQuota.daily_limit) * 100;
      
      document.getElementById('tx-quota-progress').style.width = `${txPercentage}%`;
      document.getElementById('tx-quota-text').textContent = 
        `${txQuota.daily_used}/${txQuota.daily_limit} (${txQuota.daily_remaining} remaining)`;
    }
    
    // Volume quota
    if (quotas.volume) {
      const volQuota = quotas.volume;
      const volPercentage = (volQuota.daily_used / volQuota.daily_limit) * 100;
      
      document.getElementById('vol-quota-progress').style.width = `${volPercentage}%`;
      document.getElementById('vol-quota-text').textContent = 
        `$${volQuota.daily_used.toLocaleString()}/$${volQuota.daily_limit.toLocaleString()} ($${volQuota.daily_remaining.toLocaleString()} remaining)`;
    }
  }
}

// Usage
const profileDashboard = new ProfileDashboard(apiClient);
await profileDashboard.loadProfile();
```

### Statistics Tracking

<CodeGroup>

```javascript JavaScript
class StatisticsTracker {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async getStatistics() {
    try {
      const response = await this.apiClient.get('/v1/users/profile');
      
      if (response.success) {
        return response.data.statistics;
      }
    } catch (error) {
      console.error('Failed to get statistics:', error);
    }
    return null;
  }
  
  async trackProgress() {
    const statistics = await this.getStatistics();
    
    if (statistics) {
      // Track transaction progress
      this.trackTransactionProgress(statistics);
      
      // Track volume progress
      this.trackVolumeProgress(statistics);
      
      // Track account creation progress
      this.trackAccountProgress(statistics);
    }
  }
  
  trackTransactionProgress(statistics) {
    const totalTransactions = statistics.total_transactions;
    
    // Show progress milestones
    if (totalTransactions >= 100) {
      this.showMilestone('100 transactions completed!');
    } else if (totalTransactions >= 50) {
      this.showMilestone('50 transactions completed!');
    } else if (totalTransactions >= 10) {
      this.showMilestone('10 transactions completed!');
    }
  }
  
  trackVolumeProgress(statistics) {
    const totalVolume = statistics.total_volume;
    
    // Show volume milestones
    if (totalVolume >= 100000) {
      this.showMilestone('$100,000 in total volume!');
    } else if (totalVolume >= 50000) {
      this.showMilestone('$50,000 in total volume!');
    } else if (totalVolume >= 10000) {
      this.showMilestone('$10,000 in total volume!');
    }
  }
  
  trackAccountProgress(statistics) {
    const accountsCreated = statistics.virtual_accounts_created;
    
    // Show account creation milestones
    if (accountsCreated >= 10) {
      this.showMilestone('10 virtual accounts created!');
    } else if (accountsCreated >= 5) {
      this.showMilestone('5 virtual accounts created!');
    }
  }
  
  showMilestone(message) {
    // Display milestone achievement
    console.log('Milestone achieved:', message);
    // You could show a toast notification or modal here
  }
}

// Usage
const statisticsTracker = new StatisticsTracker(apiClient);
await statisticsTracker.trackProgress();
```
