## Overview

Retrieve a list of all supported currencies with their details, including currency codes, names, symbols, and supported networks.

<Note>
This endpoint does not require authentication and can be used to build currency selection interfaces.
</Note>

## Response

<ResponseField name="currencies" type="array">
  Array of currency objects
</ResponseField>

<ResponseField name="currencies[].code" type="string">
  Currency code (e.g., `"USD"`, `"BTC"`)
</ResponseField>

<ResponseField name="currencies[].name" type="string">
  Full currency name (e.g., `"US Dollar"`, `"Bitcoin"`)
</ResponseField>

<ResponseField name="currencies[].symbol" type="string">
  Currency symbol (e.g., `"$"`, `"₿"`)
</ResponseField>

<ResponseField name="currencies[].type" type="string">
  Currency type (`fiat`, `crypto`)
</ResponseField>

<ResponseField name="currencies[].decimals" type="number">
  Number of decimal places supported
</ResponseField>

<ResponseField name="currencies[].networks" type="array">
  Supported networks for this currency
</ResponseField>

<ResponseField name="currencies[].networks[].name" type="string">
  Network name (e.g., `"Ethereum"`, `"Bitcoin"`)
</ResponseField>

<ResponseField name="currencies[].networks[].code" type="string">
  Network code (e.g., `"ETH"`, `"BTC"`)
</ResponseField>

<ResponseField name="currencies[].networks[].min_amount" type="string">
  Minimum transfer amount
</ResponseField>

<ResponseField name="currencies[].networks[].max_amount" type="string">
  Maximum transfer amount
</ResponseField>

<ResponseField name="currencies[].status" type="string">
  Currency status (`active`, `maintenance`, `deprecated`)
</ResponseField>

## Usage Examples

### Get Fiat Currencies Only

<CodeGroup>

```javascript JavaScript
async function getFiatCurrencies() {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/rates/currencies');
    const data = await response.json();
    
    return data.currencies.filter(currency => currency.type === 'fiat');
  } catch (error) {
    console.error('Error fetching fiat currencies:', error);
    throw error;
  }
}

// Usage
const fiatCurrencies = await getFiatCurrencies();
console.log('Fiat currencies:', fiatCurrencies);
```

</CodeGroup>

### Get Crypto Currencies with Networks

<CodeGroup>

```javascript JavaScript
async function getCryptoCurrencies() {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/rates/currencies');
    const data = await response.json();
    
    return data.currencies.filter(currency => 
      currency.type === 'crypto' && 
      currency.status === 'active' &&
      currency.networks.length > 0
    );
  } catch (error) {
    console.error('Error fetching crypto currencies:', error);
    throw error;
  }
}

// Usage
const cryptoCurrencies = await getCryptoCurrencies();
console.log('Crypto currencies:', cryptoCurrencies);
```

</CodeGroup>
