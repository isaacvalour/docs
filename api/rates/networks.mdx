## Overview

Retrieve a list of all supported blockchain networks with their details, including network information, supported currencies, and transfer limits.

<Note>
This endpoint does not require authentication and provides information about available networks for cryptocurrency transfers.
</Note>

## Response

<ResponseField name="networks" type="array">
  Array of network objects
</ResponseField>

<ResponseField name="networks[].id" type="string">
  Unique network identifier
</ResponseField>

<ResponseField name="networks[].name" type="string">
  Network name (e.g., `"Ethereum"`, `"Bitcoin"`)
</ResponseField>

<ResponseField name="networks[].code" type="string">
  Network code (e.g., `"ETH"`, `"BTC"`)
</ResponseField>

<ResponseField name="networks[].type" type="string">
  Network type (`mainnet`, `testnet`)
</ResponseField>

<ResponseField name="networks[].chain_id" type="number">
  Blockchain chain ID
</ResponseField>

<ResponseField name="networks[].native_currency" type="string">
  Native currency code for gas fees
</ResponseField>

<ResponseField name="networks[].supported_currencies" type="array">
  Array of supported currency codes
</ResponseField>

<ResponseField name="networks[].min_transfer" type="string">
  Minimum transfer amount
</ResponseField>

<ResponseField name="networks[].max_transfer" type="string">
  Maximum transfer amount
</ResponseField>

<ResponseField name="networks[].gas_fee_estimate" type="object">
  Gas fee estimation information
</ResponseField>

<ResponseField name="networks[].gas_fee_estimate.low" type="string">
  Low priority gas fee
</ResponseField>

<ResponseField name="networks[].gas_fee_estimate.medium" type="string">
  Medium priority gas fee
</ResponseField>

<ResponseField name="networks[].gas_fee_estimate.high" type="string">
  High priority gas fee
</ResponseField>

<ResponseField name="networks[].status" type="string">
  Network status (`active`, `maintenance`, `deprecated`)
</ResponseField>

<ResponseField name="networks[].confirmation_blocks" type="number">
  Number of confirmations required
</ResponseField>

<ResponseField name="networks[].average_confirmation_time" type="string">
  Average confirmation time in minutes
</ResponseField>

## Usage Examples

### Get Active Networks Only

<CodeGroup>

```javascript JavaScript
async function getActiveNetworks() {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/rates/networks');
    const data = await response.json();
    
    return data.networks.filter(network => network.status === 'active');
  } catch (error) {
    console.error('Error fetching active networks:', error);
    throw error;
  }
}

// Usage
const activeNetworks = await getActiveNetworks();
console.log('Active networks:', activeNetworks);
```

</CodeGroup>

### Get Networks for Specific Currency

<CodeGroup>

```javascript JavaScript
async function getNetworksForCurrency(currencyCode) {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/rates/networks');
    const data = await response.json();
    
    return data.networks.filter(network => 
      network.supported_currencies.includes(currencyCode) &&
      network.status === 'active'
    );
  } catch (error) {
    console.error('Error fetching networks for currency:', error);
    throw error;
  }
}

// Usage
const usdcNetworks = await getNetworksForCurrency('USDC');
console.log('USDC supported networks:', usdcNetworks);
```

</CodeGroup>

### Compare Network Fees

<CodeGroup>

```javascript JavaScript
async function compareNetworkFees(currencyCode) {
  try {
    const networks = await getNetworksForCurrency(currencyCode);
    
    return networks.map(network => ({
      name: network.name,
      code: network.code,
      low_fee: network.gas_fee_estimate.low,
      medium_fee: network.gas_fee_estimate.medium,
      high_fee: network.gas_fee_estimate.high,
      confirmation_time: network.average_confirmation_time
    })).sort((a, b) => parseFloat(a.medium_fee) - parseFloat(b.medium_fee));
  } catch (error) {
    console.error('Error comparing network fees:', error);
    throw error;
  }
}

// Usage
const feeComparison = await compareNetworkFees('USDC');
console.log('Network fee comparison:', feeComparison);
```

</CodeGroup>
