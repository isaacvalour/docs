## Overview

Retrieve current exchange rates for all supported currency pairs. Rates are updated in real-time and include both buy and sell rates.

<Note>
Exchange rates are valid for a limited time (typically 5 minutes). Always check the `expires_at` field before using rates for transactions.
</Note>

## Query Parameters

<ParamField query="from_currency" type="string">
  Filter rates by source currency (e.g., `"USD"`, `"NGN"`)
</ParamField>

<ParamField query="to_currency" type="string">
  Filter rates by destination currency (e.g., `"USDC"`, `"EUR"`)
</ParamField>

<ParamField query="pair" type="string">
  Filter by specific currency pair (e.g., `"USD/USDC"`)
</ParamField>

<ParamField query="include_fees" type="boolean">
  Include fee information in response (default: false)
</ParamField>

## Response

<ResponseField name="rates" type="array">
  Array of exchange rate objects
</ResponseField>

<ResponseField name="rates[].pair" type="string">
  Currency pair (e.g., `"USD/USDC"`)
</ResponseField>

<ResponseField name="rates[].from_currency" type="string">
  Source currency code
</ResponseField>

<ResponseField name="rates[].to_currency" type="string">
  Destination currency code
</ResponseField>

<ResponseField name="rates[].buy_rate" type="string">
  Rate for buying the destination currency
</ResponseField>

<ResponseField name="rates[].sell_rate" type="string">
  Rate for selling the source currency
</ResponseField>

<ResponseField name="rates[].mid_rate" type="string">
  Mid-market rate (average of buy and sell)
</ResponseField>

<ResponseField name="rates[].fees" type="object">
  Fee information (if include_fees=true)
</ResponseField>

<ResponseField name="rates[].fees.fixed" type="string">
  Fixed fee amount
</ResponseField>

<ResponseField name="rates[].fees.percentage" type="string">
  Percentage fee rate
</ResponseField>

<ResponseField name="rates[].expires_at" type="string">
  Rate expiration timestamp (ISO 8601)
</ResponseField>

<ResponseField name="rates[].updated_at" type="string">
  Last update timestamp (ISO 8601)
</ResponseField>

<ResponseField name="timestamp" type="string">
  Response timestamp (ISO 8601)
</ResponseField>

## Usage Examples

### Get Rates for Specific Pair

<CodeGroup>

```javascript JavaScript
async function getExchangeRate(fromCurrency, toCurrency) {
  try {
    const response = await fetch(`https://api.railsfromthecrypt.com/v1/rates?from_currency=${fromCurrency}&to_currency=${toCurrency}`, {
      headers: {
        'Authorization': `Bearer ${accessToken}`
      }
    });

    if (!response.ok) {
      throw new Error('Failed to fetch exchange rates');
    }

    const data = await response.json();
    
    if (data.rates.length === 0) {
      throw new Error('No rates found for this currency pair');
    }

    return data.rates[0];
  } catch (error) {
    console.error('Error fetching exchange rate:', error);
    throw error;
  }
}

// Usage
const rate = await getExchangeRate('USD', 'USDC');
console.log(`1 USD = ${rate.buy_rate} USDC`);
```

</CodeGroup>

### Check Rate Validity

<CodeGroup>

```javascript JavaScript
function isRateValid(rate) {
  const now = new Date();
  const expiresAt = new Date(rate.expires_at);
  
  return now < expiresAt;
}

async function getValidRates() {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/rates', {
      headers: {
        'Authorization': `Bearer ${accessToken}`
      }
    });

    const data = await response.json();
    
    // Filter out expired rates
    const validRates = data.rates.filter(isRateValid);
    
    return validRates;
  } catch (error) {
    console.error('Error fetching valid rates:', error);
    throw error;
  }
}
```

</CodeGroup>
