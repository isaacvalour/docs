---
title: "Webhook Logs"
api: "GET /v1/webhooks/{id}/logs"
description: "Get delivery logs for a specific webhook"
---

## Overview

Retrieve delivery logs for a specific webhook, including success/failure details and response codes.

## Path Parameters

<ParamField path="id" type="string" required>
  Unique identifier for the webhook
</ParamField>

## Query Parameters

<ParamField query="limit" type="number">
  Number of log entries to return (default: 50, max: 200)
</ParamField>

<ParamField query="offset" type="number">
  Number of log entries to skip (default: 0)
</ParamField>

<ParamField query="status" type="string">
  Filter by delivery status (`success`, `failed`)
</ParamField>

## Response

<ResponseField name="logs" type="array">
  Array of log entry objects
</ResponseField>

<ResponseField name="logs[].id" type="string">
  Log entry ID
</ResponseField>

<ResponseField name="logs[].event_type" type="string">
  Type of event that triggered the webhook
</ResponseField>

<ResponseField name="logs[].status" type="string">
  Delivery status (`success`, `failed`)
</ResponseField>

<ResponseField name="logs[].response_code" type="number">
  HTTP response code from your endpoint
</ResponseField>

<ResponseField name="logs[].response_time" type="number">
  Response time in milliseconds
</ResponseField>

<ResponseField name="logs[].attempt_count" type="number">
  Number of delivery attempts
</ResponseField>

<ResponseField name="logs[].error_message" type="string">
  Error message (if delivery failed)
</ResponseField>

<ResponseField name="logs[].timestamp" type="string">
  Delivery timestamp (ISO 8601)
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information
</ResponseField>
