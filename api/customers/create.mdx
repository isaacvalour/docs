## Overview

Create a new customer in the HR API system. This endpoint allows you to register customers for KYC/KYB verification and transaction processing.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

```javascript JavaScript
class CustomerManager {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async createCustomer(customerData) {
    try {
      const response = await this.apiClient.post('/v1/customers', customerData);
      
      if (response.success) {
        this.handleCustomerCreated(response.data);
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to create customer:', error);
      throw error;
    }
  }
  
  async createCustomerWithValidation(customerData) {
    // Validate required fields
    const validation = this.validateCustomerData(customerData);
    
    if (!validation.isValid) {
      throw new Error(`Validation failed: ${validation.errors.join(', ')}`);
    }
    
    return this.createCustomer(customerData);
  }
  
  validateCustomerData(customerData) {
    const errors = [];
    
    // Required fields validation
    if (!customerData.email) errors.push('Email is required');
    if (!customerData.first_name) errors.push('First name is required');
    if (!customerData.last_name) errors.push('Last name is required');
    if (!customerData.phone_number) errors.push('Phone number is required');
    if (!customerData.date_of_birth) errors.push('Date of birth is required');
    if (!customerData.country) errors.push('Country is required');
    if (!customerData.address) errors.push('Address is required');
    if (!customerData.document_type) errors.push('Document type is required');
    if (!customerData.document_number) errors.push('Document number is required');
    
    // Email validation
    if (customerData.email && !this.isValidEmail(customerData.email)) {
      errors.push('Invalid email format');
    }
    
    // Phone validation
    if (customerData.phone_number && !this.isValidPhone(customerData.phone_number)) {
      errors.push('Phone number must include country code');
    }
    
    // Date validation
    if (customerData.date_of_birth && !this.isValidDate(customerData.date_of_birth)) {
      errors.push('Date of birth must be in YYYY-MM-DD format');
    }
    
    // Address validation
    if (customerData.address) {
      const addressErrors = this.validateAddress(customerData.address);
      errors.push(...addressErrors);
    }
    
    // Document validation
    if (customerData.document_expiry_date && !this.isValidDate(customerData.document_expiry_date)) {
      errors.push('Document expiry date must be in YYYY-MM-DD format');
    }
    
    return {
      isValid: errors.length === 0,
      errors: errors
    };
  }
  
  validateAddress(address) {
    const errors = [];
    
    if (!address.street) errors.push('Street address is required');
    if (!address.city) errors.push('City is required');
    if (!address.state) errors.push('State is required');
    if (!address.postal_code) errors.push('Postal code is required');
    if (!address.country) errors.push('Country is required');
    
    return errors;
  }
  
  isValidEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
  }
  
  isValidPhone(phone) {
    const phoneRegex = /^\+[1-9]\d{1,14}$/;
    return phoneRegex.test(phone);
  }
  
  isValidDate(dateString) {
    const date = new Date(dateString);
    return date instanceof Date && !isNaN(date);
  }
  
  handleCustomerCreated(customerData) {
    console.log('Customer created:', customerData);
    
    // Show success message
    this.showSuccessMessage('Customer created successfully');
    
    // Display customer information
    this.displayCustomerInfo(customerData);
    
    // Trigger KYC process if needed
    if (customerData.kyc_status === 'pending') {
      this.initiateKYCProcess(customerData.id);
    }
  }
  
  displayCustomerInfo(customerData) {
    const container = document.getElementById('customer-info');
    
    container.innerHTML = `
      <div class="customer-card">
        <div class="customer-header">
          <h3>Customer Created</h3>
          <span class="status-badge ${customerData.status}">${customerData.status}</span>
        </div>
        
        <div class="customer-details">
          <div class="detail-section">
            <h4>Personal Information</h4>
            <div class="detail-grid">
              <div class="detail-item">
                <label>Customer ID:</label>
                <span class="customer-id">${customerData.id}</span>
              </div>
              <div class="detail-item">
                <label>Name:</label>
                <span>${customerData.first_name} ${customerData.last_name}</span>
              </div>
              <div class="detail-item">
                <label>Email:</label>
                <span>${customerData.email}</span>
              </div>
              <div class="detail-item">
                <label>Phone:</label>
                <span>${customerData.phone_number}</span>
              </div>
              <div class="detail-item">
                <label>Date of Birth:</label>
                <span>${customerData.date_of_birth}</span>
              </div>
              <div class="detail-item">
                <label>Country:</label>
                <span>${customerData.country}</span>
              </div>
            </div>
          </div>
          
          <div class="detail-section">
            <h4>Address</h4>
            <div class="address-info">
              <div>${customerData.address.street}</div>
              <div>${customerData.address.city}, ${customerData.address.state} ${customerData.address.postal_code}</div>
              <div>${customerData.address.country}</div>
            </div>
          </div>
          
          <div class="detail-section">
            <h4>Document Information</h4>
            <div class="detail-grid">
              <div class="detail-item">
                <label>Document Type:</label>
                <span>${customerData.document_type}</span>
              </div>
              <div class="detail-item">
                <label>Document Number:</label>
                <span>${customerData.document_number}</span>
              </div>
              <div class="detail-item">
                <label>Issued Date:</label>
                <span>${customerData.document_issued_date}</span>
              </div>
              <div class="detail-item">
                <label>Expiry Date:</label>
                <span>${customerData.document_expiry_date || 'N/A'}</span>
              </div>
              <div class="detail-item">
                <label>Issuing Authority:</label>
                <span>${customerData.document_issuing_authority}</span>
              </div>
            </div>
          </div>
          
          <div class="detail-section">
            <h4>Verification Status</h4>
            <div class="verification-info">
              <div class="verification-item">
                <label>KYC Status:</label>
                <span class="kyc-status ${customerData.kyc_status}">${customerData.kyc_status}</span>
              </div>
              <div class="verification-item">
                <label>KYC Level:</label>
                <span>${customerData.kyc_level}</span>
              </div>
              <div class="verification-item">
                <label>Risk Score:</label>
                <span class="risk-score">${customerData.risk_score}/100</span>
              </div>
            </div>
          </div>
        </div>
      </div>
    `;
  }
  
  initiateKYCProcess(customerId) {
    // This would typically trigger the KYC verification process
    console.log('Initiating KYC process for customer:', customerId);
    
    // Show KYC initiation message
    this.showInfoMessage('KYC verification process initiated. Customer will receive instructions via email.');
  }
  
  showSuccessMessage(message) {
    const toast = document.createElement('div');
    toast.className = 'toast success';
    toast.textContent = message;
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.remove();
    }, 3000);
  }
  
  showInfoMessage(message) {
    const toast = document.createElement('div');
    toast.className = 'toast info';
    toast.textContent = message;
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.remove();
    }, 3000);
  }
  
  showErrorMessage(message) {
    const toast = document.createElement('div');
    toast.className = 'toast error';
    toast.textContent = message;
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.remove();
    }, 3000);
  }
}

// Usage

    address: {

      city: formData.get('city'),
      state: formData.get('state'),
      postal_code: formData.get('postal_code'),
      country: formData.get('country')
    },
    document_type: formData.get('document_type'),
    document_number: formData.get('document_number'),
    document_issued_date: formData.get('document_issued_date'),
    document_expiry_date: formData.get('document_expiry_date'),
    document_issuing_authority: formData.get('document_issuing_authority'),
    occupation: formData.get('occupation'),
    employer: formData.get('employer'),
    annual_income: parseFloat(formData.get('annual_income')),
    source_of_funds: formData.get('source_of_funds'),
    purpose_of_account: formData.get('purpose_of_account'),
    metadata: {
      referral_code: formData.get('referral_code'),
      campaign_id: formData.get('campaign_id'),

  };
  
  try {
    const customer = await customerManager.createCustomerWithValidation(customerData);
```
