title: "Get Customer Verification Tasks"
api: "GET /v1/customers/{id}/verification-tasks"
description: "Get verification tasks for a specific customer"
---

## Overview

Retrieve all verification tasks associated with a specific customer. This endpoint provides detailed information about KYC/KYB verification tasks, their status, and requirements.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

<ParamField path="id" type="string" required>
  Customer ID.
</ParamField>

<ParamField query="status" type="string">
  Filter by task status. Possible values: "pending", "in_progress", "completed", "failed", "cancelled".
</ParamField>

<ParamField query="type" type="string">
  Filter by task type. Possible values: "document_verification", "address_verification", "identity_verification", "biometric_verification", "phone_verification", "email_verification".
</ParamField>

<ParamField query="priority" type="string">
  Filter by task priority. Possible values: "low", "medium", "high", "urgent".
</ParamField>

<ParamField query="assigned_to" type="string">
  Filter by assigned user ID.
</ParamField>

<ParamField query="created_after" type="string">
  Filter tasks created after this date (ISO 8601 format).
</ParamField>

<ParamField query="created_before" type="string">
  Filter tasks created before this date (ISO 8601 format).
</ParamField>

<ParamField query="due_after" type="string">
  Filter tasks due after this date (ISO 8601 format).
</ParamField>

<ParamField query="due_before" type="string">
  Filter tasks due before this date (ISO 8601 format).
</ParamField>

<ParamField query="sort" type="string">
  Sort order. Possible values: "created_at", "-created_at", "due_date", "-due_date", "priority", "-priority". Default: "-created_at".
</ParamField>

<ParamField query="include_details" type="boolean">
  Include detailed task information. Default: false.
</ParamField>

## Implementation Example

```javascript JavaScript
class CustomerVerificationTasks {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async getVerificationTasks(customerId, options = {}) {
    try {
      const params = new URLSearchParams();
      
      if (options.status) params.append('status', options.status);
      if (options.type) params.append('type', options.type);
      if (options.priority) params.append('priority', options.priority);
      if (options.assignedTo) params.append('assigned_to', options.assignedTo);
      if (options.createdAfter) params.append('created_after', options.createdAfter);
      if (options.createdBefore) params.append('created_before', options.createdBefore);
      if (options.dueAfter) params.append('due_after', options.dueAfter);
      if (options.dueBefore) params.append('due_before', options.dueBefore);
      if (options.sort) params.append('sort', options.sort);
      if (options.includeDetails) params.append('include_details', 'true');
      
      const response = await this.apiClient.get(`/v1/customers/${customerId}/verification-tasks?${params}`);
      
      if (response.success) {
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to get verification tasks:', error);
      throw error;
    }
  }
  
  async getPendingTasks(customerId) {
    return this.getVerificationTasks(customerId, { status: 'pending' });
  }
  
  async getCompletedTasks(customerId) {
    return this.getVerificationTasks(customerId, { status: 'completed' });
  }
  
  async getFailedTasks(customerId) {
    return this.getVerificationTasks(customerId, { status: 'failed' });
  }
  
  async getTasksByType(customerId, taskType) {
    return this.getVerificationTasks(customerId, { type: taskType });
  }
  
  async getOverdueTasks(customerId) {
    const now = new Date().toISOString();
    return this.getVerificationTasks(customerId, { 
      dueBefore: now,
      status: 'pending'
    });
  }
  
  async getTasksWithDetails(customerId) {
    return this.getVerificationTasks(customerId, { includeDetails: true });
  }
  
  displayVerificationTasks(tasksData) {
    const container = document.getElementById('verification-tasks');
    
    container.innerHTML = `
      <div class="verification-tasks-container">
        <div class="tasks-header">
          <h3>Verification Tasks</h3>
          <div class="tasks-summary">
            ${this.renderTasksSummary(tasksData.summary)}
          </div>
        </div>
        
        <div class="tasks-filters">
          ${this.renderTasksFilters()}
        </div>
        
        <div class="tasks-list">
          ${tasksData.tasks.map(task => this.renderTask(task)).join('')}
        </div>
      </div>
    `;
  }
  
  renderTasksSummary(summary) {
    return `
      <div class="summary-grid">
        <div class="summary-item">
          <span class="summary-label">Total Tasks:</span>
          <span class="summary-value">${summary.total_tasks}</span>
        </div>
        <div class="summary-item">
          <span class="summary-label">Completed:</span>
          <span class="summary-value success">${summary.completed_tasks}</span>
        </div>
        <div class="summary-item">
          <span class="summary-label">Pending:</span>
          <span class="summary-value warning">${summary.pending_tasks}</span>
        </div>
        <div class="summary-item">
          <span class="summary-label">Failed:</span>
          <span class="summary-value error">${summary.failed_tasks}</span>
        </div>
        <div class="summary-item">
          <span class="summary-label">Overdue:</span>
          <span class="summary-value error">${summary.overdue_tasks}</span>
        </div>
        <div class="summary-item">
          <span class="summary-label">Completion Rate:</span>
          <span class="summary-value">${summary.completion_rate.toFixed(1)}%</span>
        </div>
        <div class="summary-item">
          <span class="summary-label">Avg. Time:</span>
          <span class="summary-value">${summary.average_completion_time.toFixed(1)}h</span>
        </div>
      </div>
    `;
  }
  
  renderTasksFilters() {
    return `
      <div class="filters-container">
        <div class="filter-group">
          <label>Status:</label>
          <select id="status-filter" onchange="filterTasks()">
            <option value="">All Statuses</option>
            <option value="pending">Pending</option>
            <option value="in_progress">In Progress</option>
            <option value="completed">Completed</option>
            <option value="failed">Failed</option>
            <option value="cancelled">Cancelled</option>
          </select>
        </div>
        
        <div class="filter-group">
          <label>Type:</label>
          <select id="type-filter" onchange="filterTasks()">
            <option value="">All Types</option>
            <option value="document_verification">Document Verification</option>
            <option value="address_verification">Address Verification</option>
            <option value="identity_verification">Identity Verification</option>
            <option value="biometric_verification">Biometric Verification</option>
            <option value="phone_verification">Phone Verification</option>
            <option value="email_verification">Email Verification</option>
          </select>
        </div>
        
        <div class="filter-group">
          <label>Priority:</label>
          <select id="priority-filter" onchange="filterTasks()">
            <option value="">All Priorities</option>
            <option value="low">Low</option>
            <option value="medium">Medium</option>
            <option value="high">High</option>
            <option value="urgent">Urgent</option>
          </select>
        </div>
        
        <div class="filter-group">
          <button class="btn btn-primary" onclick="refreshTasks()">Refresh</button>
        </div>
      </div>
    `;
  }
  
  renderTask(task) {
    const statusColor = this.getTaskStatusColor(task.status);
    const priorityColor = this.getPriorityColor(task.priority);
    const isOverdue = this.isTaskOverdue(task);
    
    return `
      <div class="task-card ${isOverdue ? 'overdue' : ''}" data-task-id="${task.id}">
        <div class="task-header">
          <div class="task-title">
            <h4>${task.title}</h4>
            <span class="task-type">${this.formatTaskType(task.type)}</span>
          </div>
          <div class="task-status">
            <span class="status-badge ${statusColor}">${task.status}</span>
            <span class="priority-badge ${priorityColor}">${task.priority}</span>
          </div>
        </div>
        
        <div class="task-content">
          <div class="task-description">
            ${task.description}
          </div>
          
          <div class="task-details">
            <div class="detail-row">
              <label>Assigned to:</label>
              <span>${task.assigned_to_name || 'Unassigned'}</span>
            </div>
            <div class="detail-row">
              <label>Due Date:</label>
              <span class="${isOverdue ? 'overdue' : ''}">${new Date(task.due_date).toLocaleDateString()}</span>
            </div>
            <div class="detail-row">
              <label>Progress:</label>
              <div class="progress-bar">
                <div class="progress-fill" style="width: ${task.progress}%"></div>
              </div>
              <span>${task.progress}%</span>
            </div>
            <div class="detail-row">
              <label>Estimated Duration:</label>
              <span>${task.estimated_duration} minutes</span>
            </div>
          </div>
          
          ${task.requirements ? this.renderTaskRequirements(task.requirements) : ''}
          ${task.attachments ? this.renderTaskAttachments(task.attachments) : ''}
          ${task.verification_result ? this.renderVerificationResult(task.verification_result) : ''}
          
          ${task.notes ? `
            <div class="task-notes">
              <label>Notes:</label>
              <p>${task.notes}</p>
            </div>
          ` : ''}
        </div>
        
        <div class="task-actions">
          <button class="btn btn-primary" onclick="viewTask('${task.id}')">
            View Details
          </button>
          ${task.status === 'pending' ? `
            <button class="btn btn-success" onclick="startTask('${task.id}')">
              Start Task
            </button>
          ` : ''}
          ${task.status === 'in_progress' ? `
            <button class="btn btn-warning" onclick="completeTask('${task.id}')">
              Complete Task
            </button>
          ` : ''}
          ${task.status === 'failed' ? `
            <button class="btn btn-info" onclick="retryTask('${task.id}')">
              Retry Task
            </button>
          ` : ''}
        </div>
        
        <div class="task-meta">
          <span class="created-date">Created: ${new Date(task.created_at).toLocaleDateString()}</span>
          <span class="updated-date">Updated: ${new Date(task.updated_at).toLocaleDateString()}</span>
          ${task.completed_at ? `
            <span class="completed-date">Completed: ${new Date(task.completed_at).toLocaleDateString()}</span>
          ` : ''}
        </div>
      </div>
    `;
  }
  
  renderTaskRequirements(requirements) {
    return `
      <div class="task-requirements">
        <h5>Requirements:</h5>
        <ul class="requirements-list">
          ${requirements.map(req => `
            <li class="requirement-item ${req.required ? 'required' : 'optional'}">
              <span class="requirement-type">${req.type}</span>
              <span class="requirement-description">${req.description}</span>
              <span class="requirement-status">${req.required ? 'Required' : 'Optional'}</span>
            </li>
          `).join('')}
        </ul>
      </div>
    `;
  }
  
  renderTaskAttachments(attachments) {
    return `
      <div class="task-attachments">
        <h5>Attachments:</h5>
        <div class="attachments-list">
          ${attachments.map(attachment => `
            <div class="attachment-item">
              <div class="attachment-info">
                <span class="attachment-name">${attachment.name}</span>
                <span class="attachment-type">${attachment.type}</span>
                <span class="attachment-date">${new Date(attachment.uploaded_at).toLocaleDateString()}</span>
              </div>
              <div class="attachment-actions">
                <button class="btn btn-sm btn-secondary" onclick="viewAttachment('${attachment.id}')">
                  View
                </button>
                <button class="btn btn-sm btn-primary" onclick="downloadAttachment('${attachment.id}')">
                  Download
                </button>
              </div>
            </div>
          `).join('')}
        </div>
      </div>
    `;
  }
  
  renderVerificationResult(result) {
    const resultColor = this.getResultColor(result.status);
    
    return `
      <div class="verification-result">
        <h5>Verification Result:</h5>
        <div class="result-details">
          <div class="result-status">
            <span class="status-badge ${resultColor}">${result.status}</span>
            <span class="confidence-score">Confidence: ${result.score}%</span>
          </div>
          
          ${result.details ? `
            <div class="result-details-list">
              ${Object.entries(result.details).map(([key, value]) => `
                <div class="detail-item">
                  <label>${key.replace(/_/g, ' ')}:</label>
                  <span>${value}</span>
                </div>
              `).join('')}
            </div>
          ` : ''}
          
          ${result.reviewer_notes ? `
            <div class="reviewer-notes">
              <label>Reviewer Notes:</label>
              <p>${result.reviewer_notes}</p>
            </div>
          ` : ''}
          
          <div class="result-meta">
            <span>Reviewed by: ${result.reviewed_by}</span>
            <span>Reviewed at: ${new Date(result.reviewed_at).toLocaleString()}</span>
          </div>
        </div>
      </div>
    `;
  }
  
  getTaskStatusColor(status) {
    const colors = {
      'pending': 'warning',
      'in_progress': 'info',
      'completed': 'success',
      'failed': 'error',
      'cancelled': 'secondary'
    };
    return colors[status] || 'default';
  }
  
  getPriorityColor(priority) {
    const colors = {
      'low': 'success',
      'medium': 'warning',
      'high': 'error',
      'urgent': 'danger'
    };
    return colors[priority] || 'default';
  }
  
  getResultColor(status) {
    const colors = {
      'approved': 'success',
      'rejected': 'error',
      'pending': 'warning',
      'incomplete': 'info'
    };
    return colors[status] || 'default';
  }
  
  formatTaskType(type) {
    return type.replace(/_/g, ' ').replace(/\b\w/g, l => l.toUpperCase());
  }
  
  isTaskOverdue(task) {
    if (task.status === 'completed' || task.status === 'cancelled') {
      return false;
    }
    
    const dueDate = new Date(task.due_date);
    const now = new Date();
    
    return dueDate < now;
  }
  
  generateTasksReport(tasksData) {
    const report = {
      customer_id: tasksData.customer_id,
      summary: tasksData.summary,
      tasks: tasksData.tasks,
      generated_at: new Date().toISOString(),
      analysis: this.analyzeTasks(tasksData.tasks)
    };
    
    return report;
  }
  
  analyzeTasks(tasks) {
    const analysis = {
      task_types: {},
      completion_times: [],
      failure_reasons: [],
      overdue_tasks: []
    };
    
    tasks.forEach(task => {
      // Count by type
      analysis.task_types[task.type] = (analysis.task_types[task.type] || 0) + 1;
      
      // Calculate completion times
      if (task.completed_at) {
        const created = new Date(task.created_at);
        const completed = new Date(task.completed_at);
        const duration = (completed - created) / (1000 * 60 * 60); // hours
        analysis.completion_times.push(duration);
      }
      
      // Track failures
      if (task.status === 'failed') {
        analysis.failure_reasons.push(task.notes || 'No reason provided');
      }
      
      // Track overdue tasks
      if (this.isTaskOverdue(task)) {
        analysis.overdue_tasks.push({
          id: task.id,
          title: task.title,
          due_date: task.due_date,
          days_overdue: Math.ceil((new Date() - new Date(task.due_date)) / (1000 * 60 * 60 * 24))
        });
      }
    });
    
    return analysis;
  }
```
