## Overview

Retrieve detailed information about a specific customer by their ID. This endpoint provides comprehensive customer data including personal information, verification status, and transaction history.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

<ParamField path="id" type="string" required>
  Customer ID.
</ParamField>

<ParamField query="include_verification_tasks" type="boolean">
  Include verification tasks in the response. Default: false.
</ParamField>

<ParamField query="include_transaction_history" type="boolean">
  Include recent transaction history. Default: false.
</ParamField>

<ParamField query="include_risk_assessment" type="boolean">
  Include detailed risk assessment. Default: false.
</ParamField>

<ParamField query="include_metadata" type="boolean">
  Include customer metadata. Default: false.
</ParamField>

```bash cURL
curl -X GET "https://api.railsfromthecrypt.com/v1/customers/cust_1234567890?include_verification_tasks=true&include_transaction_history=true&include_risk_assessment=true" \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.railsfromthecrypt.com/v1/customers/cust_1234567890?include_verification_tasks=true&include_transaction_history=true&include_risk_assessment=true', {
  headers: {
    'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
  }
});

const data = await response.json();
console.log(data);
```

```python Python
import requests

response = requests.get(
    'https://api.railsfromthecrypt.com/v1/customers/cust_1234567890',
    headers={
        'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
    },
    params={
        'include_verification_tasks': True,
        'include_transaction_history': True,
        'include_risk_assessment': True
    }
)

data = response.json()
print(data)
```

```javascript JavaScript
class CustomerDetailManager {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async getCustomer(customerId, options = {}) {
    try {
      const params = new URLSearchParams();
      
      if (options.includeVerificationTasks) params.append('include_verification_tasks', 'true');
      if (options.includeTransactionHistory) params.append('include_transaction_history', 'true');
      if (options.includeRiskAssessment) params.append('include_risk_assessment', 'true');
      if (options.includeMetadata) params.append('include_metadata', 'true');
      
      const response = await this.apiClient.get(`/v1/customers/${customerId}?${params}`);
      
      if (response.success) {
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to get customer:', error);
      throw error;
    }
  }
  
  async getCustomerBasic(customerId) {
    return this.getCustomer(customerId);
  }
  
  async getCustomerWithTasks(customerId) {
    return this.getCustomer(customerId, { includeVerificationTasks: true });
  }
  
  async getCustomerWithHistory(customerId) {
    return this.getCustomer(customerId, { includeTransactionHistory: true });
  }
  
  async getCustomerWithRiskAssessment(customerId) {
    return this.getCustomer(customerId, { includeRiskAssessment: true });
  }
  
  async getCustomerFull(customerId) {
    return this.getCustomer(customerId, {
      includeVerificationTasks: true,
      includeTransactionHistory: true,
      includeRiskAssessment: true,
      includeMetadata: true
    });
  }
  
  displayCustomer(customerData) {
    const container = document.getElementById('customer-details');
    
    container.innerHTML = `
      <div class="customer-detail-card">
        <div class="customer-header">
          <div class="customer-info">
            <h2>${customerData.first_name} ${customerData.last_name}</h2>
            <p class="customer-email">${customerData.email}</p>
            <p class="customer-id">ID: ${customerData.id}</p>
          </div>
          <div class="customer-status">
            <span class="status-badge ${this.getStatusColor(customerData.status)}">${customerData.status}</span>
            <span class="kyc-badge ${this.getKYCColor(customerData.kyc_status)}">${customerData.kyc_status}</span>
            <span class="risk-badge ${this.getRiskColor(customerData.risk_score)}">Risk: ${customerData.risk_score}/100</span>
          </div>
        </div>
        
        <div class="customer-content">
          <div class="customer-sections">
            ${this.renderPersonalInfo(customerData)}
            ${this.renderAddressInfo(customerData)}
            ${this.renderDocumentInfo(customerData)}
            ${this.renderEmploymentInfo(customerData)}
            ${this.renderVerificationInfo(customerData)}
            ${customerData.verification_tasks ? this.renderVerificationTasks(customerData.verification_tasks) : ''}
            ${customerData.transaction_history ? this.renderTransactionHistory(customerData.transaction_history) : ''}
            ${customerData.risk_assessment ? this.renderRiskAssessment(customerData.risk_assessment) : ''}
            ${customerData.metadata ? this.renderMetadata(customerData.metadata) : ''}
          </div>

        <div class="customer-actions">
          <button class="btn btn-primary" onclick="editCustomer('${customerData.id}')">
            Edit Customer
          </button>
          <button class="btn btn-secondary" onclick="viewTransactions('${customerData.id}')">
            View All Transactions
          </button>
          ${customerData.kyc_status === 'pending' ? `
            <button class="btn btn-warning" onclick="verifyCustomer('${customerData.id}')">
              Verify Customer
            </button>
          ` : ''}
          <button class="btn btn-danger" onclick="suspendCustomer('${customerData.id}')">
```
