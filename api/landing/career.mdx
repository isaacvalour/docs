## Overview

Submit a career application form for job opportunities at HashRails. This endpoint is used for job applications and recruitment.

<Note>
This endpoint does not require authentication and is typically used on public career pages.
</Note>

## Request Body

<ParamField body="first_name" type="string" required>
  First name of the applicant
</ParamField>

<ParamField body="last_name" type="string" required>
  Last name of the applicant
</ParamField>

<ParamField body="email" type="string" required>
  Email address for contact
</ParamField>

<ParamField body="phone" type="string">
  Phone number for contact
</ParamField>

<ParamField body="position" type="string" required>
  Position being applied for
</ParamField>

<ParamField body="experience_level" type="string">
  Experience level (`entry`, `mid`, `senior`, `lead`, `executive`)
</ParamField>

<ParamField body="resume_url" type="string">
  URL to resume or CV
</ParamField>

<ParamField body="portfolio_url" type="string">
  URL to portfolio or GitHub profile
</ParamField>

<ParamField body="cover_letter" type="string">
  Cover letter text
</ParamField>

<ParamField body="availability" type="string">
  Availability for interview (`immediate`, `2_weeks`, `1_month`, `flexible`)
</ParamField>

<ParamField body="salary_expectation" type="string">
  Salary expectation range
</ParamField>

<ParamField body="location" type="string">
  Preferred work location
</ParamField>

<ParamField body="remote_ok" type="boolean">
  Whether remote work is acceptable
</ParamField>

<ParamField body="referral_source" type="string">
  How they heard about the position
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the application was submitted successfully
</ResponseField>

<ResponseField name="message" type="string">
  Confirmation message
</ResponseField>

<ResponseField name="application_id" type="string">
  Unique identifier for the application
</ResponseField>

<ResponseField name="submitted_at" type="string">
  Submission timestamp (ISO 8601)
</ResponseField>

<ResponseField name="next_steps" type="string">
  Information about next steps
</ResponseField>

## Usage Examples

### Submit Career Application

<CodeGroup>

```javascript JavaScript
async function submitCareerApplication(formData) {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/landing/career', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(formData)
    });

    const result = await response.json();
    
    if (result.success) {
      // Show success message
      showSuccessMessage('Application submitted successfully!');
      
      // Redirect to thank you page
      redirectToThankYouPage(result.application_id);
    } else {
      // Show error message
      showErrorMessage('Failed to submit application. Please try again.');
    }
    
    return result;
  } catch (error) {
    console.error('Error submitting application:', error);
    showErrorMessage('An error occurred. Please try again.');
    throw error;
  }
}

// Usage
const applicationData = {
  first_name: 'Jane',
  last_name: 'Doe',
  email: 'jane.doe@email.com',
  phone: '+1234567890',
  position: 'Senior Software Engineer',
  experience_level: 'senior',
  resume_url: 'https://example.com/resume.pdf',
  portfolio_url: 'https://github.com/janedoe',
  cover_letter: 'I am excited about the opportunity to work at HashRails...',
  availability: '2_weeks',
  salary_expectation: '$80,000 - $100,000',
  location: 'San Francisco, CA',
  remote_ok: true,
  referral_source: 'Company website'
};

await submitCareerApplication(applicationData);
```

</CodeGroup>
