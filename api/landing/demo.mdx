## Overview

Submit a demo request form to schedule a demonstration of the HR API platform. This endpoint is used for lead generation and demo scheduling.

<Note>
This endpoint does not require authentication and is typically used on public landing pages.
</Note>

## Request Body

<ParamField body="name" type="string" required>
  Full name of the person requesting the demo
</ParamField>

<ParamField body="email" type="string" required>
  Email address for demo scheduling
</ParamField>

<ParamField body="company" type="string" required>
  Company name
</ParamField>

<ParamField body="phone" type="string">
  Phone number for contact
</ParamField>

<ParamField body="role" type="string">
  Job role or title
</ParamField>

<ParamField body="company_size" type="string">
  Company size (`startup`, `small`, `medium`, `large`, `enterprise`)
</ParamField>

<ParamField body="use_case" type="string">
  Primary use case for the API
</ParamField>

<ParamField body="message" type="string">
  Additional message or requirements
</ParamField>

<ParamField body="preferred_time" type="string">
  Preferred time for demo (ISO 8601)
</ParamField>

<ParamField body="timezone" type="string">
  Timezone for scheduling
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the demo request was submitted successfully
</ResponseField>

<ResponseField name="message" type="string">
  Confirmation message
</ResponseField>

<ResponseField name="demo_id" type="string">
  Unique identifier for the demo request
</ResponseField>

<ResponseField name="submitted_at" type="string">
  Submission timestamp (ISO 8601)
</ResponseField>

<ResponseField name="next_steps" type="string">
  Information about next steps
</ResponseField>

## Usage Examples

### Submit Demo Request Form

<CodeGroup>

```javascript JavaScript
async function submitDemoRequest(formData) {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/landing/demo', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(formData)
    });

    const result = await response.json();
    
    if (result.success) {
      // Show success message
      showSuccessMessage('Demo request submitted successfully!');
      
      // Redirect to thank you page or show confirmation
      redirectToThankYouPage(result.demo_id);
    } else {
      // Show error message
      showErrorMessage('Failed to submit demo request. Please try again.');
    }
    
    return result;
  } catch (error) {
    console.error('Error submitting demo request:', error);
    showErrorMessage('An error occurred. Please try again.');
    throw error;
  }
}

// Usage
const formData = {
  name: 'John Smith',
  email: 'john.smith@company.com',
  company: 'Acme Corp',
  phone: '+1234567890',
  role: 'CTO',
  company_size: 'medium',
  use_case: 'Payment processing for e-commerce',
  message: 'Looking to integrate payment processing for our online store'
};

await submitDemoRequest(formData);
```

</CodeGroup>
