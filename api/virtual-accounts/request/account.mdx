title: "Get Virtual Account"
api: "GET /v1/virtual-accounts/request/{id}/account"
description: "Get virtual account details by request ID"
---

## Overview

Get the virtual account details for a specific virtual account request. This endpoint returns the complete virtual account information including account numbers, bank details, and status.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

<ParamField path="id" type="string" required>
  Virtual account request ID.
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful.
</ResponseField>

<ResponseField name="data" type="object">
  Virtual account data.
</ResponseField>

<ResponseField name="data.id" type="string">
  Virtual account request ID.
</ResponseField>

<ResponseField name="data.currency" type="string">
  Virtual account currency.
</ResponseField>

<ResponseField name="data.status" type="string">
  Virtual account request status.
</ResponseField>

<ResponseField name="data.virtualAccount" type="object">
  Virtual account details.
</ResponseField>

<ResponseField name="data.virtualAccount.id" type="string">
  Virtual account ID.
</ResponseField>

<ResponseField name="data.virtualAccount.account_number" type="string">
  Virtual account number.
</ResponseField>

<ResponseField name="data.virtualAccount.account_name" type="string">
  Virtual account name.
</ResponseField>

<ResponseField name="data.virtualAccount.bank_name" type="string">
  Bank name.
</ResponseField>

<ResponseField name="data.virtualAccount.routing_number" type="string">
  Bank routing number (USD accounts only).
</ResponseField>

<ResponseField name="data.virtualAccount.reference" type="string">
  Unique reference for tracking payments.
</ResponseField>

<ResponseField name="data.virtualAccount.provider" type="string">
  Payment provider (e.g., "Bridge", "Nomba", "Juicyway").
</ResponseField>

<ResponseField name="data.virtualAccount.network" type="string">
  Blockchain network (for crypto settlement).
</ResponseField>

<ResponseField name="data.virtualAccount.wallet_address" type="string">
  Settlement wallet address.
</ResponseField>

<ResponseField name="data.virtualAccount.created_at" type="string">
  Virtual account creation timestamp.
</ResponseField>

<ResponseField name="data.virtualAccount.updated_at" type="string">
  Last update timestamp.
</ResponseField>

<ResponseField name="data.created_at" type="string">
  Request creation timestamp.
</ResponseField>

<ResponseField name="data.updated_at" type="string">
  Request last update timestamp.
</ResponseField>

## Implementation Example

<CodeGroup>

```javascript JavaScript
class VirtualAccountManager {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async getVirtualAccount(requestId) {
    try {
      const response = await this.apiClient.get(`/v1/virtual-accounts/request/${requestId}/account`);
      
      if (response.success) {
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to get virtual account:', error);
      throw error;
    }
  }
  
  displayVirtualAccount(accountData) {
    const container = document.getElementById('virtual-account-details');
    
    if (accountData.virtualAccount) {
      const account = accountData.virtualAccount;
      
      container.innerHTML = `
        <div class="virtual-account-card">
          <h3>Virtual Account Details</h3>
          <div class="account-info">
            <div class="info-row">
              <label>Account Number:</label>
              <span class="account-number">${account.account_number}</span>
              <button onclick="copyToClipboard('${account.account_number}')" class="copy-btn">Copy</button>
            </div>
            <div class="info-row">
              <label>Account Name:</label>
              <span>${account.account_name}</span>
            </div>
            <div class="info-row">
              <label>Bank Name:</label>
              <span>${account.bank_name}</span>
            </div>
            ${account.routing_number ? `
              <div class="info-row">
                <label>Routing Number:</label>
                <span class="routing-number">${account.routing_number}</span>
                <button onclick="copyToClipboard('${account.routing_number}')" class="copy-btn">Copy</button>
              </div>
            ` : ''}
            <div class="info-row">
              <label>Reference:</label>
              <span class="reference">${account.reference}</span>
              <button onclick="copyToClipboard('${account.reference}')" class="copy-btn">Copy</button>
            </div>
            <div class="info-row">
              <label>Provider:</label>
              <span>${account.provider}</span>
            </div>
            <div class="info-row">
              <label>Network:</label>
              <span>${account.network}</span>
            </div>
            <div class="info-row">
              <label>Wallet Address:</label>
              <span class="wallet-address">${account.wallet_address}</span>
              <button onclick="copyToClipboard('${account.wallet_address}')" class="copy-btn">Copy</button>
            </div>
          </div>
        </div>
      `;
    } else {
      container.innerHTML = `
        <div class="virtual-account-card">
          <h3>Virtual Account Status</h3>
          <div class="status-info">
            <p><strong>Status:</strong> ${accountData.status}</p>
            <p><strong>Currency:</strong> ${accountData.currency}</p>
            <p>Virtual account is not yet created. Please wait for approval.</p>
          </div>
        </div>
      `;
    }
  }
  
  generatePaymentInstructions(accountData) {
    if (!accountData.virtualAccount) {
      return null;
    }
    
    const account = accountData.virtualAccount;
    const currency = accountData.currency;
    
    const instructions = {
      USD: {
        title: 'Payment Instructions for USD',
        steps: [
          `Account Number: ${account.account_number}`,
          `Account Name: ${account.account_name}`,
          `Bank Name: ${account.bank_name}`,
          `Routing Number: ${account.routing_number}`,
          `Reference: ${account.reference}`
        ],
        note: 'Use ACH transfer or wire transfer. Include the reference number in your payment.'
      },
      NGN: {
        title: 'Payment Instructions for NGN',
        steps: [
          `Account Number: ${account.account_number}`,
          `Account Name: ${account.account_name}`,
          `Bank Name: ${account.bank_name}`,
          `Reference: ${account.reference}`
        ],
        note: 'Use local bank transfer. Include the reference number in your payment.'
      }
    };
    
    return instructions[currency];
  }
  
  displayPaymentInstructions(accountData) {
    const instructions = this.generatePaymentInstructions(accountData);
    
    if (!instructions) {
      return;
    }
    
    const container = document.getElementById('payment-instructions');
    
    container.innerHTML = `
      <div class="payment-instructions-card">
        <h3>${instructions.title}</h3>
        <div class="instructions-list">
          ${instructions.steps.map(step => `<div class="instruction-step">${step}</div>`).join('')}
        </div>
        <div class="instruction-note">
          <p><strong>Note:</strong> ${instructions.note}</p>
        </div>
      </div>
    `;
  }
}

// Usage
const virtualAccountManager = new VirtualAccountManager(apiClient);

// Get virtual account details
document.getElementById('get-account-details').addEventListener('click', async () => {
  const requestId = 'req_1234567890';
  
  try {
    const accountData = await virtualAccountManager.getVirtualAccount(requestId);
    virtualAccountManager.displayVirtualAccount(accountData);
    virtualAccountManager.displayPaymentInstructions(accountData);
  } catch (error) {
    console.error('Failed to get virtual account:', error);
  }
});

// Copy to clipboard function
function copyToClipboard(text) {
  navigator.clipboard.writeText(text).then(() => {
    // Show success message
    const toast = document.createElement('div');
    toast.className = 'toast success';
    toast.textContent = 'Copied to clipboard!';
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.remove();
    }, 2000);
  });
}
```
