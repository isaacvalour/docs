---

## Overview

Suspend a virtual account to temporarily disable it. This endpoint allows you to suspend virtual accounts for security, compliance, or administrative reasons.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

<ParamField path="id" type="string" required>
  Virtual account ID to suspend.
</ParamField>

<ParamField body="reason" type="string" required>
  Reason for suspending the virtual account.
</ParamField>

<ParamField body="notes" type="string">
  Additional notes about the suspension.
</ParamField>

<ParamField body="metadata" type="object">
  Additional metadata for the suspension.
</ParamField>

<ParamField body="metadata.suspension_type" type="string">
  Type of suspension. Possible values: "temporary", "permanent", "security", "compliance".
</ParamField>

<ParamField body="metadata.duration" type="number">
  Duration of suspension in hours (for temporary suspensions).
</ParamField>

<ParamField body="metadata.notify_client" type="boolean">
  Whether to notify the client about the suspension.
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the virtual account was suspended successfully.
</ResponseField>

<ResponseField name="message" type="string">
  Success message.
</ResponseField>

<ResponseField name="data" type="object">
  Suspended virtual account data.
</ResponseField>

<ResponseField name="data.id" type="string">
  Virtual account ID.
</ResponseField>

<ResponseField name="data.account_number" type="string">
  Virtual account number.
</ResponseField>

<ResponseField name="data.account_name" type="string">
  Virtual account name.
</ResponseField>

<ResponseField name="data.bank_name" type="string">
  Bank name.
</ResponseField>

<ResponseField name="data.status" type="string">
  Updated virtual account status.
</ResponseField>

<ResponseField name="data.suspended_at" type="string">
  Timestamp when the account was suspended.
</ResponseField>

<ResponseField name="data.suspended_by" type="string">
  User who suspended the account.
</ResponseField>

<ResponseField name="data.suspension_reason" type="string">
  Reason for suspension.
</ResponseField>

<ResponseField name="data.suspension_notes" type="string">
  Notes about the suspension.
</ResponseField>

<ResponseField name="data.suspension_metadata" type="object">
  Suspension metadata.
</ResponseField>

<ResponseField name="data.suspension_metadata.suspension_type" type="string">
  Type of suspension.
</ResponseField>

<ResponseField name="data.suspension_metadata.duration" type="number">
  Duration of suspension in hours.
</ResponseField>

<ResponseField name="data.suspension_metadata.expires_at" type="string">
  When the suspension expires (for temporary suspensions).
</ResponseField>

<ResponseField name="data.created_at" type="string">
  Virtual account creation timestamp.
</ResponseField>

<ResponseField name="data.updated_at" type="string">
  Last update timestamp.
</ResponseField>

## Example Request

<CodeGroup>

```bash cURL
curl -X POST "https://api.railsfromthecrypt.com/v1/virtual-accounts/va_1234567890/suspend" \

    "notes": "Multiple failed login attempts detected",
    "metadata": {
      "suspension_type": "security",
      "duration": 24,
      "notify_client": true
```

## Implementation Example

<CodeGroup>

```javascript JavaScript
class VirtualAccountSuspension {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async suspendAccount(accountId, reason, options = {}) {
    try {
      const response = await this.apiClient.post(`/v1/virtual-accounts/${accountId}/suspend`, {
        reason: reason,
        notes: options.notes,
        metadata: options.metadata
      });
      
      if (response.success) {
        this.handleSuspendedAccount(response.data);
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to suspend account:', error);
      throw error;
    }
  }
  
  async suspendForSecurity(accountId, reason, duration = 24) {
    return this.suspendAccount(accountId, reason, {
      notes: 'Suspended for security reasons',
      metadata: {
        suspension_type: 'security',
        duration: duration,
        notify_client: true
      }
    });
  }
  
  async suspendForCompliance(accountId, reason) {
    return this.suspendAccount(accountId, reason, {
      notes: 'Suspended for compliance reasons',
      metadata: {
        suspension_type: 'compliance',
        notify_client: true
      }
    });
  }
  
  async suspendTemporarily(accountId, reason, duration) {
    return this.suspendAccount(accountId, reason, {
      notes: 'Temporarily suspended',
      metadata: {
        suspension_type: 'temporary',
        duration: duration,
        notify_client: true
      }
    });
  }
  
  async suspendPermanently(accountId, reason) {
    return this.suspendAccount(accountId, reason, {
      notes: 'Permanently suspended',
      metadata: {
        suspension_type: 'permanent',
        notify_client: true
      }
    });
  }
  
  handleSuspendedAccount(accountData) {
    console.log('Account suspended:', accountData);
    
    // Show success message
    this.showSuccessMessage('Virtual account suspended successfully');
    
    // Update UI to show suspended status
    this.updateAccountStatus(accountData);
    
    // Send notification if required
    if (accountData.suspension_metadata?.notify_client) {
      this.sendSuspensionNotification(accountData);
    }
    
    // Log suspension for audit
    this.logSuspension(accountData);
  }
  
  updateAccountStatus(accountData) {
    const statusElement = document.getElementById(`account-status-${accountData.id}`);
    if (statusElement) {
      statusElement.textContent = 'Suspended';
      statusElement.className = 'status suspended';
    }
    
    // Disable account actions
    const actionButtons = document.querySelectorAll(`[data-account-id="${accountData.id}"]`);
    actionButtons.forEach(button => {
      button.disabled = true;
      button.textContent = 'Account Suspended';
    });
  }
  
  sendSuspensionNotification(accountData) {
    const notificationData = {
      type: 'virtual_account_suspended',
      account_id: accountData.id,
      account_number: accountData.account_number,
      reason: accountData.suspension_reason,
      notes: accountData.suspension_notes,
      suspension_type: accountData.suspension_metadata.suspension_type,
      expires_at: accountData.suspension_metadata.expires_at,
      message: 'Your virtual account has been suspended.'
    };
    
    // This would typically call a notification service
    console.log('Sending suspension notification:', notificationData);
  }
  
  logSuspension(accountData) {
    const logEntry = {
      action: 'suspend_virtual_account',
      account_id: accountData.id,
      account_number: accountData.account_number,
      reason: accountData.suspension_reason,
      notes: accountData.suspension_notes,
      suspension_type: accountData.suspension_metadata.suspension_type,
      duration: accountData.suspension_metadata.duration,
      suspended_by: accountData.suspended_by,
      suspended_at: accountData.suspended_at,
      timestamp: new Date().toISOString()
    };
    
    // This would typically call a logging service
    console.log('Suspension logged:', logEntry);
  }
  
  showSuccessMessage(message) {
    const toast = document.createElement('div');
    toast.className = 'toast success';
    toast.textContent = message;
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.remove();
    }, 3000);
  }
  
  showErrorMessage(message) {
    const toast = document.createElement('div');
    toast.className = 'toast error';
    toast.textContent = message;
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.remove();
    }, 3000);
  }
}

// Usage
const suspensionManager = new VirtualAccountSuspension(apiClient);

// Suspend for security reasons
document.getElementById('suspend-security').addEventListener('click', async () => {
  const accountId = 'va_1234567890';
  const reason = 'Suspicious activity detected';
  const duration = 24; // 24 hours
  
  try {
    const result = await suspensionManager.suspendForSecurity(accountId, reason, duration);
    console.log('Account suspended for security:', result);
  } catch (error) {
    console.error('Failed to suspend account:', error);
  }
});

// Suspend for compliance reasons
document.getElementById('suspend-compliance').addEventListener('click', async () => {
  const accountId = 'va_1234567890';
  const reason = 'Compliance review required';
  
  try {
    const result = await suspensionManager.suspendForCompliance(accountId, reason);
    console.log('Account suspended for compliance:', result);
  } catch (error) {
    console.error('Failed to suspend account:', error);
  }
});

// Suspend temporarily
document.getElementById('suspend-temporary').addEventListener('click', async () => {
  const accountId = 'va_1234567890';
  const reason = 'Temporary suspension for review';
  const duration = 48; // 48 hours
  
  try {
    const result = await suspensionManager.suspendTemporarily(accountId, reason, duration);
    console.log('Account temporarily suspended:', result);
  } catch (error) {
    console.error('Failed to suspend account:', error);
  }
});
```
