title: "Get Virtual Account"
api: "GET /v1/virtual-accounts/{id}"
description: "Get virtual account details by ID"
---

## Overview

Get detailed information about a specific virtual account by its ID. This endpoint returns comprehensive virtual account information including account details, status, and configuration.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

<ParamField path="id" type="string" required>
  Virtual account ID.
</ParamField>

<ParamField query="include_activities" type="boolean">
  Include recent activities in the response. Default: false.
</ParamField>

<ParamField query="include_metrics" type="boolean">
  Include basic metrics in the response. Default: false.
</ParamField>

## Example Request

<CodeGroup>

```bash cURL
curl -X GET "https://api.railsfromthecrypt.com/v1/virtual-accounts/va_1234567890?include_activities=true&include_metrics=true" \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.railsfromthecrypt.com/v1/virtual-accounts/va_1234567890?include_activities=true&include_metrics=true', {
  headers: {
    'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
  }
});

const data = await response.json();
console.log(data);
```

```python Python
import requests

response = requests.get(
    'https://api.railsfromthecrypt.com/v1/virtual-accounts/va_1234567890',
    headers={
        'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
    },
    params={
        'include_activities': True,
        'include_metrics': True
    }
)

data = response.json()
print(data)
```

## Implementation Example

<CodeGroup>

```javascript JavaScript
class VirtualAccountManager {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async getVirtualAccount(accountId, options = {}) {
    try {
      const params = new URLSearchParams();
      
      if (options.includeActivities) params.append('include_activities', 'true');
      if (options.includeMetrics) params.append('include_metrics', 'true');
      
      const response = await this.apiClient.get(`/v1/virtual-accounts/${accountId}?${params}`);
      
      if (response.success) {
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to get virtual account:', error);
      throw error;
    }
  }
  
  async getAccountDetails(accountId) {
    return this.getVirtualAccount(accountId, {
      includeActivities: false,
      includeMetrics: false
    });
  }
  
  async getAccountWithActivities(accountId) {
    return this.getVirtualAccount(accountId, {
      includeActivities: true,
      includeMetrics: false
    });
  }
  
  async getAccountWithMetrics(accountId) {
    return this.getVirtualAccount(accountId, {
      includeActivities: false,
      includeMetrics: true
    });
  }
  
  async getFullAccountInfo(accountId) {
    return this.getVirtualAccount(accountId, {
      includeActivities: true,
      includeMetrics: true
    });
  }
  
  displayVirtualAccount(accountData) {
    const container = document.getElementById('virtual-account-details');
    
    container.innerHTML = `
      <div class="virtual-account-card">
        <div class="account-header">
          <h3>Virtual Account Details</h3>
          <span class="status-badge ${accountData.status}">${accountData.status}</span>
        </div>
        
        <div class="account-info">
          <div class="info-section">
            <h4>Account Information</h4>
            <div class="info-grid">
              <div class="info-item">
                <label>Account Number:</label>
                <span class="account-number">${accountData.account_number}</span>
                <button onclick="copyToClipboard('${accountData.account_number}')" class="copy-btn">Copy</button>
              </div>
              <div class="info-item">
                <label>Account Name:</label>
                <span>${accountData.account_name}</span>
              </div>
              <div class="info-item">
                <label>Bank Name:</label>
                <span>${accountData.bank_name}</span>
              </div>
              ${accountData.routing_number ? `
                <div class="info-item">
                  <label>Routing Number:</label>
                  <span class="routing-number">${accountData.routing_number}</span>
                  <button onclick="copyToClipboard('${accountData.routing_number}')" class="copy-btn">Copy</button>
                </div>
              ` : ''}
              <div class="info-item">
                <label>Reference:</label>
                <span class="reference">${accountData.reference}</span>
                <button onclick="copyToClipboard('${accountData.reference}')" class="copy-btn">Copy</button>
              </div>
              <div class="info-item">
                <label>Provider:</label>
                <span>${accountData.provider}</span>
              </div>
              <div class="info-item">
                <label>Network:</label>
                <span>${accountData.network}</span>
              </div>
              <div class="info-item">
                <label>Currency:</label>
                <span>${accountData.currency}</span>
              </div>
            </div>
          </div>
          
          <div class="info-section">
            <h4>Configuration</h4>
            <div class="config-grid">
              <div class="config-item">
                <label>Auto Settlement:</label>
                <span class="config-value ${accountData.configuration.auto_settlement ? 'enabled' : 'disabled'}">
                  ${accountData.configuration.auto_settlement ? 'Enabled' : 'Disabled'}
                </span>
              </div>
              <div class="config-item">
                <label>Settlement Threshold:</label>
                <span>${accountData.currency} ${accountData.configuration.settlement_threshold.toLocaleString()}</span>
              </div>
              <div class="config-item">
                <label>Webhook URL:</label>
                <span class="webhook-url">${accountData.configuration.webhook_url || 'Not set'}</span>
              </div>
            </div>
          </div>
          
          <div class="info-section">
            <h4>Limits</h4>
            <div class="limits-grid">
              <div class="limit-item">
                <label>Daily Limit:</label>
                <span>${accountData.currency} ${accountData.limits.daily_limit.toLocaleString()}</span>
              </div>
              <div class="limit-item">
                <label>Monthly Limit:</label>
                <span>${accountData.currency} ${accountData.limits.monthly_limit.toLocaleString()}</span>
              </div>
              <div class="limit-item">
                <label>Max Transaction:</label>
                <span>${accountData.currency} ${accountData.limits.max_transaction.toLocaleString()}</span>
              </div>
              <div class="limit-item">
                <label>Min Transaction:</label>
                <span>${accountData.currency} ${accountData.limits.min_transaction.toLocaleString()}</span>
              </div>
            </div>
          </div>
          
          ${accountData.activities ? `
            <div class="info-section">
              <h4>Recent Activities</h4>
              <div class="activities-list">
                ${accountData.activities.map(activity => `
                  <div class="activity-item">
                    <div class="activity-type">${activity.type}</div>
                    <div class="activity-description">${activity.description}</div>
                    ${activity.amount ? `<div class="activity-amount">${accountData.currency} ${activity.amount.toLocaleString()}</div>` : ''}
                    <div class="activity-status ${activity.status}">${activity.status}</div>
                    <div class="activity-time">${new Date(activity.created_at).toLocaleString()}</div>
                  </div>
                `).join('')}
              </div>
            </div>
          ` : ''}
          
          ${accountData.metrics ? `
            <div class="info-section">
              <h4>Basic Metrics</h4>
              <div class="metrics-grid">
                <div class="metric-item">
                  <label>Total Transactions:</label>
                  <span>${accountData.metrics.total_transactions.toLocaleString()}</span>
                </div>
                <div class="metric-item">
                  <label>Total Volume:</label>
                  <span>${accountData.currency} ${accountData.metrics.total_volume.toLocaleString()}</span>
                </div>
                <div class="metric-item">
                  <label>Success Rate:</label>
                  <span>${accountData.metrics.success_rate.toFixed(1)}%</span>
                </div>
                <div class="metric-item">
                  <label>Last Transaction:</label>
                  <span>${new Date(accountData.metrics.last_transaction_at).toLocaleString()}</span>
                </div>
              </div>
            </div>
          ` : ''}
        </div>
      </div>
    `;
  }
  
  generatePaymentInstructions(accountData) {
    const instructions = {
      USD: {
        title: 'Payment Instructions for USD',
        steps: [
          `Account Number: ${accountData.account_number}`,
          `Account Name: ${accountData.account_name}`,
          `Bank Name: ${accountData.bank_name}`,
          `Routing Number: ${accountData.routing_number}`,
          `Reference: ${accountData.reference}`
        ],
        note: 'Use ACH transfer or wire transfer. Include the reference number in your payment.'
      },
      NGN: {
        title: 'Payment Instructions for NGN',
        steps: [
          `Account Number: ${accountData.account_number}`,
          `Account Name: ${accountData.account_name}`,
          `Bank Name: ${accountData.bank_name}`,
          `Reference: ${accountData.reference}`
        ],
        note: 'Use local bank transfer. Include the reference number in your payment.'
      }
    };
    
    return instructions[accountData.currency];
  }
  
  displayPaymentInstructions(accountData) {
    const instructions = this.generatePaymentInstructions(accountData);
    
    if (!instructions) {
      return;
    }
    
    const container = document.getElementById('payment-instructions');
    
    container.innerHTML = `
      <div class="payment-instructions-card">
        <h3>${instructions.title}</h3>
        <div class="instructions-list">
          ${instructions.steps.map(step => `<div class="instruction-step">${step}</div>`).join('')}
        </div>
        <div class="instruction-note">
          <p><strong>Note:</strong> ${instructions.note}</p>
        </div>
      </div>
    `;
  }
}

// Usage
const virtualAccountManager = new VirtualAccountManager(apiClient);

// Get basic account details
document.getElementById('get-account-details').addEventListener('click', async () => {
  const accountId = 'va_1234567890';
  
  try {
    const accountData = await virtualAccountManager.getAccountDetails(accountId);
    virtualAccountManager.displayVirtualAccount(accountData);
    virtualAccountManager.displayPaymentInstructions(accountData);
  } catch (error) {
    console.error('Failed to get account details:', error);
  }
});

// Get account with activities
document.getElementById('get-account-activities').addEventListener('click', async () => {
  const accountId = 'va_1234567890';
  
  try {
    const accountData = await virtualAccountManager.getAccountWithActivities(accountId);
    virtualAccountManager.displayVirtualAccount(accountData);
  } catch (error) {
    console.error('Failed to get account activities:', error);
  }
});

// Get full account info
document.getElementById('get-full-account-info').addEventListener('click', async () => {
  const accountId = 'va_1234567890';
  
  try {
    const accountData = await virtualAccountManager.getFullAccountInfo(accountId);
    virtualAccountManager.displayVirtualAccount(accountData);
  } catch (error) {
    console.error('Failed to get full account info:', error);
  }
});

function copyToClipboard(text) {
  navigator.clipboard.writeText(text).then(() => {
    const toast = document.createElement('div');
    toast.className = 'toast success';
    toast.textContent = 'Copied to clipboard!';
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.remove();
    }, 2000);
  });
}
```
