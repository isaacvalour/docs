## Overview

Check current system limits and quotas for the authenticated user. This endpoint provides information about rate limits, usage quotas, and remaining allowances.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

## Usage Examples

### Check Rate Limits

<CodeGroup>

```javascript JavaScript
async function checkRateLimits() {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/system/limit-check', {
      headers: {
        'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
      }
    });
    
    const data = await response.json();
    
    if (data.success) {
      const apiLimits = data.data.rate_limits.api_requests;
      const webhookLimits = data.data.rate_limits.webhook_requests;
      
      console.log(`API Requests: ${apiLimits.remaining}/${apiLimits.limit} remaining`);
      console.log(`Webhook Requests: ${webhookLimits.remaining}/${webhookLimits.limit} remaining`);
      
      // Check if approaching limits
      if (apiLimits.remaining < 100) {
        console.warn('API rate limit approaching');
      }
      
      if (webhookLimits.remaining < 100) {
        console.warn('Webhook rate limit approaching');
      }
    }
  } catch (error) {
    console.error('Failed to check rate limits:', error);
  }
}
```

</CodeGroup>

### Check Quotas

<CodeGroup>

```javascript JavaScript
async function checkQuotas() {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/system/limit-check', {
      headers: {
        'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
      }
    });
    
    const data = await response.json();
    
    if (data.success) {
      const quotas = data.data.quotas;
      
      // Check virtual account quota
      if (quotas.virtual_accounts.remaining === 0) {
        console.warn('Virtual account quota exceeded');
        return false;
      }
      
      // Check transaction quota
      if (quotas.transactions.daily_remaining === 0) {
        console.warn('Daily transaction quota exceeded');
        return false;
      }
      
      // Check volume quota
      if (quotas.volume.daily_remaining === 0) {
        console.warn('Daily volume quota exceeded');
        return false;
      }
      
      return true;
    }
  } catch (error) {
    console.error('Failed to check quotas:', error);
    return false;
  }
}
```

</CodeGroup>

### Monitor Limits

<CodeGroup>

```javascript JavaScript
class LimitMonitor {
  constructor(apiClient) {
    this.apiClient = apiClient;
    this.limits = null;
  }
  
  async updateLimits() {
    try {
      const response = await this.apiClient.get('/v1/system/limit-check');
      this.limits = response.data;
      return this.limits;
    } catch (error) {
      console.error('Failed to update limits:', error);
      return null;
    }
  }
  
  isApiLimitReached() {
    if (!this.limits) return false;
    return this.limits.rate_limits.api_requests.remaining === 0;
  }
  
  isWebhookLimitReached() {
    if (!this.limits) return false;
    return this.limits.rate_limits.webhook_requests.remaining === 0;
  }
  
  isVirtualAccountQuotaReached() {
    if (!this.limits) return false;
    return this.limits.quotas.virtual_accounts.remaining === 0;
  }
  
  isTransactionQuotaReached() {
    if (!this.limits) return false;
    return this.limits.quotas.transactions.daily_remaining === 0;
  }
  
  isVolumeQuotaReached() {
    if (!this.limits) return false;
    return this.limits.quotas.volume.daily_remaining === 0;
  }
  
  getApiLimitPercentage() {
    if (!this.limits) return 0;
    const apiLimits = this.limits.rate_limits.api_requests;
    return ((apiLimits.limit - apiLimits.remaining) / apiLimits.limit) * 100;
  }
  
  getTransactionQuotaPercentage() {
    if (!this.limits) return 0;
    const txQuotas = this.limits.quotas.transactions;
    return (txQuotas.daily_used / txQuotas.daily_limit) * 100;
  }
}

// Usage
const monitor = new LimitMonitor(apiClient);

// Update limits every 5 minutes
setInterval(async () => {
  await monitor.updateLimits();
  
  if (monitor.isApiLimitReached()) {
    console.warn('API rate limit reached');
  }
  
  if (monitor.isTransactionQuotaReached()) {
    console.warn('Transaction quota reached');
  }
}, 300000);
```

</CodeGroup>
