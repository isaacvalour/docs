## Overview

Create a new onboarding session for a user. This endpoint initiates the onboarding process and returns a unique onboarding token that can be used to track progress and complete the onboarding flow.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

<ParamField body="type" type="string" required>
  Type of onboarding. Possible values: "individual", "business", "enterprise".
</ParamField>

<ParamField body="step" type="string">
  Initial step to start from. Defaults to "welcome".
</ParamField>

<ParamField body="metadata" type="object">
  Additional metadata for the onboarding session.
</ParamField>

<ParamField body="metadata.source" type="string">
  Source of the onboarding (e.g., "web", "mobile", "api").
</ParamField>

<ParamField body="metadata.referral_code" type="string">
  Referral code if applicable.
</ParamField>

<ParamField body="metadata.campaign" type="string">
  Marketing campaign identifier.
</ParamField>

## Implementation Example

<CodeGroup>

```javascript JavaScript
class OnboardingManager {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async createOnboarding(type, options = {}) {
    try {
      const response = await this.apiClient.post('/v1/onboard', {
        type: type,
        step: options.step || 'welcome',
        metadata: options.metadata || {}
      });
      
      if (response.success) {
        this.storeOnboardingSession(response.data);
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to create onboarding session:', error);
      throw error;
    }
  }
  
  storeOnboardingSession(sessionData) {
    // Store session data in localStorage
    localStorage.setItem('onboarding_session', JSON.stringify(sessionData));
    
    // Store token for API calls
    localStorage.setItem('onboarding_token', sessionData.token);
  }
  
  getOnboardingSession() {
    const sessionData = localStorage.getItem('onboarding_session');
    return sessionData ? JSON.parse(sessionData) : null;
  }
  
  getOnboardingToken() {
    return localStorage.getItem('onboarding_token');
  }
  
  clearOnboardingSession() {
    localStorage.removeItem('onboarding_session');
    localStorage.removeItem('onboarding_token');
  }
  
  async startIndividualOnboarding() {
    return this.createOnboarding('individual', {
      metadata: {
        source: 'web',
        campaign: 'organic'
      }
    });
  }
  
  async startBusinessOnboarding() {
    return this.createOnboarding('business', {
      metadata: {
        source: 'web',
        campaign: 'business_signup'
      }
    });
  }
  
  async startEnterpriseOnboarding() {
    return this.createOnboarding('enterprise', {
      metadata: {
        source: 'web',
        campaign: 'enterprise_signup'
      }
    });
  }
}

// Usage
const onboardingManager = new OnboardingManager(apiClient);

// Start individual onboarding
document.getElementById('start-individual').addEventListener('click', async () => {
  try {
    const session = await onboardingManager.startIndividualOnboarding();
    console.log('Onboarding started:', session);
    
    // Redirect to onboarding flow
    window.location.href = `/onboarding/${session.id}`;
  } catch (error) {
    console.error('Failed to start onboarding:', error);
  }
});

// Start business onboarding
document.getElementById('start-business').addEventListener('click', async () => {
  try {
    const session = await onboardingManager.startBusinessOnboarding();
    console.log('Business onboarding started:', session);
    
    // Redirect to business onboarding flow
    window.location.href = `/onboarding/business/${session.id}`;
  } catch (error) {
    console.error('Failed to start business onboarding:', error);
  }
});
```
