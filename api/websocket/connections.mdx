---

## Overview

Retrieve information about active WebSocket connections, including connection details, user information, and connection statistics.

<Note>
This endpoint requires authentication and provides information about current WebSocket connections.
</Note>

## Query Parameters

<ParamField query="limit" type="number">
  Number of connections to return (default: 50, max: 200)
</ParamField>

<ParamField query="offset" type="number">
  Number of connections to skip (default: 0)
</ParamField>

<ParamField query="user_id" type="string">
  Filter connections by user ID
</ParamField>

<ParamField query="status" type="string">
  Filter by connection status (`active`, `idle`, `disconnected`)
</ParamField>

<ParamField query="sort" type="string">
  Sort order (`connected_at`, `-connected_at`, `user_id`, `-user_id`)
</ParamField>

## Response

<ResponseField name="connections" type="array">
  Array of connection objects
</ResponseField>

<ResponseField name="connections[].id" type="string">
  Unique connection identifier
</ResponseField>

<ResponseField name="connections[].user_id" type="string">
  User ID associated with the connection
</ResponseField>

<ResponseField name="connections[].ip_address" type="string">
  IP address of the connection
</ResponseField>

<ResponseField name="connections[].user_agent" type="string">
  User agent string
</ResponseField>

<ResponseField name="connections[].status" type="string">
  Connection status (`active`, `idle`, `disconnected`)
</ResponseField>

<ResponseField name="connections[].connected_at" type="string">
  Connection timestamp (ISO 8601)
</ResponseField>

<ResponseField name="connections[].last_activity" type="string">
  Last activity timestamp (ISO 8601)
</ResponseField>

<ResponseField name="connections[].subscriptions" type="array">
  Array of active subscriptions
</ResponseField>

<ResponseField name="connections[].subscriptions[].channel" type="string">
  Subscription channel name
</ResponseField>

<ResponseField name="connections[].subscriptions[].subscribed_at" type="string">
  Subscription timestamp (ISO 8601)
</ResponseField>

<ResponseField name="connections[].message_count" type="number">
  Total messages sent to this connection
</ResponseField>

<ResponseField name="connections[].error_count" type="number">
  Total errors for this connection
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information
</ResponseField>
