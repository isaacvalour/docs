---

## Overview

Retrieve information about WebSocket subscriptions, available channels, and subscription statistics.

<Note>
This endpoint requires authentication and provides information about WebSocket subscription channels and their usage.
</Note>

## Query Parameters

<ParamField query="limit" type="number">
  Number of subscriptions to return (default: 50, max: 200)
</ParamField>

<ParamField query="offset" type="number">
  Number of subscriptions to skip (default: 0)
</ParamField>

<ParamField query="channel" type="string">
  Filter by channel name
</ParamField>

<ParamField query="user_id" type="string">
  Filter by user ID
</ParamField>

<ParamField query="status" type="string">
  Filter by subscription status (`active`, `inactive`)
</ParamField>

## Response

<ResponseField name="subscriptions" type="array">
  Array of subscription objects
</ResponseField>

<ResponseField name="subscriptions[].id" type="string">
  Unique subscription identifier
</ResponseField>

<ResponseField name="subscriptions[].user_id" type="string">
  User ID associated with the subscription
</ResponseField>

<ResponseField name="subscriptions[].channel" type="string">
  Subscription channel name
</ResponseField>

<ResponseField name="subscriptions[].status" type="string">
  Subscription status (`active`, `inactive`)
</ResponseField>

<ResponseField name="subscriptions[].subscribed_at" type="string">
  Subscription timestamp (ISO 8601)
</ResponseField>

<ResponseField name="subscriptions[].last_message" type="string">
  Last message timestamp (ISO 8601)
</ResponseField>

<ResponseField name="subscriptions[].message_count" type="number">
  Total messages received for this subscription
</ResponseField>

<ResponseField name="subscriptions[].error_count" type="number">
  Total errors for this subscription
</ResponseField>

<ResponseField name="channels" type="array">
  Array of available channels
</ResponseField>

<ResponseField name="channels[].name" type="string">
  Channel name
</ResponseField>

<ResponseField name="channels[].description" type="string">
  Channel description
</ResponseField>

<ResponseField name="channels[].subscriber_count" type="number">
  Number of active subscribers
</ResponseField>

<ResponseField name="channels[].message_rate" type="number">
  Messages per minute
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information
</ResponseField>
