## Overview

Retrieve detailed information about a specific transaction using its unique internal ID. This endpoint is useful for system-to-system communication and when you have the internal transaction ID.

<Note>
The internal ID is a UUID that's used for system operations. For user-facing applications, consider using the reference ID endpoint instead.
</Note>

## Path Parameters

<ParamField path="id" type="string" required>
  The unique internal ID of the transaction to retrieve
  
  **Format:** UUID (e.g., `123e4567-e89b-12d3-a456-426614174000`)
</ParamField>

## Usage Examples

### Get Transaction Details

<CodeGroup>

```javascript JavaScript
async function getTransactionById(transactionId) {
  try {
    const response = await fetch(`https://api.railsfromthecrypt.com/v1/transactions/${transactionId}`, {
      headers: {
        'Authorization': `Bearer ${accessToken}`
      }
    });

    if (!response.ok) {
      if (response.status === 404) {
        throw new Error('Transaction not found');
      }
      throw new Error('Failed to fetch transaction');
    }

    const transaction = await response.json();
    
    console.log(`Transaction ${transactionId} details:`, transaction);
    
    return transaction;
  } catch (error) {
    console.error('Error fetching transaction:', error);
    throw error;
  }
}
```

```python Python
def get_transaction_by_id(transaction_id):
    try:
        response = requests.get(
            f'https://api.railsfromthecrypt.com/v1/transactions/{transaction_id}',
            headers={'Authorization': f'Bearer {access_token}'}
        )
        
        if not response.ok:
            if response.status_code == 404:
                raise Exception('Transaction not found')
            raise Exception('Failed to fetch transaction')
        
        transaction = response.json()
        
        print(f'Transaction {transaction_id} details:', transaction)
        
        return transaction
        
    except Exception as e:
        print(f'Error fetching transaction: {e}')
        raise
```

</CodeGroup>

### Validate Transaction Access

<CodeGroup>

```javascript JavaScript
async function validateTransactionAccess(transactionId, userId) {
  try {
    const transaction = await getTransactionById(transactionId);
    
    // Check if user has access to this transaction
    if (transaction.user_id !== userId) {
      throw new Error('Access denied: Transaction belongs to different user');
    }
    
    return transaction;
  } catch (error) {
    console.error('Error validating transaction access:', error);
    throw error;
  }
}
```

</CodeGroup>

### Get Transaction with Compliance Info

<CodeGroup>

```javascript JavaScript
async function getTransactionWithCompliance(transactionId) {
  try {
    const transaction = await getTransactionById(transactionId);
    
    // Check compliance status
    const compliance = transaction.compliance || {};
    
    const complianceStatus = {
      aml: compliance.aml_status || 'unknown',
      kyc: compliance.kyc_status || 'unknown',
      sanctions: compliance.sanctions_check || 'unknown'
    };
    
    // Determine if transaction is compliant
    const isCompliant = complianceStatus.aml === 'passed' && 
                       complianceStatus.kyc === 'verified' && 
                       complianceStatus.sanctions === 'clear';
    
    return {
      ...transaction,
      complianceStatus,
      isCompliant
    };
  } catch (error) {
    console.error('Error getting transaction with compliance:', error);
    throw error;
  }
}
```

</CodeGroup>

### Batch Transaction Lookup

<CodeGroup>

```javascript JavaScript
async function getMultipleTransactions(transactionIds) {
  try {
    const promises = transactionIds.map(id => getTransactionById(id));
    const transactions = await Promise.allSettled(promises);
    
    const results = {
      successful: [],
      failed: []
    };
    
    transactions.forEach((result, index) => {
      if (result.status === 'fulfilled') {
        results.successful.push(result.value);
      } else {
        results.failed.push({
          id: transactionIds[index],
          error: result.reason.message
        });
      }
    });
    
    return results;
  } catch (error) {
    console.error('Error getting multiple transactions:', error);
    throw error;
  }
}

// Usage
const transactionIds = [
  '123e4567-e89b-12d3-a456-426614174000',
  '123e4567-e89b-12d3-a456-426614174001',
  '123e4567-e89b-12d3-a456-426614174002'
];

const results = await getMultipleTransactions(transactionIds);
console.log('Successful:', results.successful.length);
console.log('Failed:', results.failed.length);
```

### Transaction Audit Trail

<CodeGroup>

```javascript JavaScript
async function getTransactionAuditTrail(transactionId) {
  try {
    const transaction = await getTransactionById(transactionId);
    
    const auditTrail = {
      transactionId: transaction.id,
      reference: transaction.reference,
      status: transaction.status,
      createdAt: transaction.created_at,
      updatedAt: transaction.updated_at,
      events: transaction.events || [],
      metadata: transaction.metadata || {},
      compliance: transaction.compliance || {}
    };
    
    // Add event timeline
    auditTrail.timeline = transaction.events.map(event => ({
      timestamp: event.timestamp,
      status: event.status,
      description: event.description,
      duration: calculateDuration(event.timestamp, transaction.updated_at)
    }));
    
    return auditTrail;
  } catch (error) {
    console.error('Error getting transaction audit trail:', error);
    throw error;
  }
}

function calculateDuration(startTime, endTime) {
  const start = new Date(startTime);
  const end = new Date(endTime);
  const diffMs = end - start;
  const diffMinutes = Math.floor(diffMs / 60000);
  const diffSeconds = Math.floor((diffMs % 60000) / 1000);
  
  return `${diffMinutes}m ${diffSeconds}s`;
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="ID Management">
    - Store transaction IDs securely in your system
    - Use UUIDs for internal operations
    - Implement proper error handling for invalid IDs
    - Cache transaction data to reduce API calls
  </Accordion>
  <Accordion title="Access Control">
    - Always validate user access to transactions
    - Implement proper authorization checks
    - Log access attempts for audit purposes
    - Use reference IDs for user-facing operations
  </Accordion>
  <Accordion title="Error Handling">
    - Handle 404 errors gracefully (transaction not found)
    - Implement retry logic for network errors
    - Show appropriate error messages to users
    - Log errors for debugging purposes
  </Accordion>
  <Accordion title="Performance">
    - Cache transaction data when appropriate
    - Use batch operations for multiple lookups
    - Implement request deduplication
    - Monitor API usage and rate limits
  </Accordion>
</AccordionGroup>
