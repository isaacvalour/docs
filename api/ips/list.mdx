## Overview

Retrieve a list of all whitelisted IP addresses with optional filtering and pagination.

## Query Parameters

<ParamField query="limit" type="number">
  Number of IPs to return (default: 20, max: 100)
</ParamField>

<ParamField query="offset" type="number">
  Number of IPs to skip (default: 0)
</ParamField>

<ParamField query="status" type="string">
  Filter by status (`active`, `inactive`)
</ParamField>

<ParamField query="search" type="string">
  Search by IP address or label
</ParamField>

## Response

<ResponseField name="ip_addresses" type="array">
  Array of IP address objects
</ResponseField>

<ResponseField name="ip_addresses[].id" type="string">
  IP address ID
</ResponseField>

<ResponseField name="ip_addresses[].ip_address" type="string">
  Whitelisted IP address or range
</ResponseField>

<ResponseField name="ip_addresses[].label" type="string">
  IP address label
</ResponseField>

<ResponseField name="ip_addresses[].status" type="string">
  IP address status
</ResponseField>

<ResponseField name="ip_addresses[].created_at" type="string">
  Creation timestamp (ISO 8601)
</ResponseField>

<ResponseField name="ip_addresses[].last_used" type="string">
  Last usage timestamp (ISO 8601)
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information
</ResponseField>
