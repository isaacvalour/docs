---
title: "List Notifications"
api: "GET /v1/notifications"
description: "Get list of all notifications for the authenticated user"
---

## Overview

Retrieve a paginated list of all notifications for the authenticated user with optional filtering by status and type.

## Query Parameters

<ParamField query="limit" type="number">
  Number of notifications to return (default: 20, max: 100)
</ParamField>

<ParamField query="offset" type="number">
  Number of notifications to skip (default: 0)
</ParamField>

<ParamField query="status" type="string">
  Filter by notification status (`unread`, `read`, `archived`)
</ParamField>

<ParamField query="type" type="string">
  Filter by notification type (`transaction`, `system`, `security`, `marketing`)
</ParamField>

<ParamField query="created_after" type="string">
  Filter notifications created after this date (ISO 8601)
</ParamField>

<ParamField query="created_before" type="string">
  Filter notifications created before this date (ISO 8601)
</ParamField>

<ParamField query="sort" type="string">
  Sort order (`created_at`, `-created_at`, `priority`, `-priority`)
</ParamField>

## Response

<ResponseField name="notifications" type="array">
  Array of notification objects
</ResponseField>

<ResponseField name="notifications[].id" type="string">
  Unique identifier for the notification
</ResponseField>

<ResponseField name="notifications[].title" type="string">
  Notification title
</ResponseField>

<ResponseField name="notifications[].message" type="string">
  Notification message content
</ResponseField>

<ResponseField name="notifications[].type" type="string">
  Notification type (`transaction`, `system`, `security`, `marketing`)
</ResponseField>

<ResponseField name="notifications[].status" type="string">
  Notification status (`unread`, `read`, `archived`)
</ResponseField>

<ResponseField name="notifications[].priority" type="string">
  Notification priority (`low`, `medium`, `high`, `urgent`)
</ResponseField>

<ResponseField name="notifications[].data" type="object">
  Additional notification data
</ResponseField>

<ResponseField name="notifications[].created_at" type="string">
  Creation timestamp (ISO 8601)
</ResponseField>

<ResponseField name="notifications[].read_at" type="string">
  Read timestamp (ISO 8601)
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information
</ResponseField>

## Example Request

<CodeGroup>

```bash cURL
# Get all notifications
curl -X GET "https://api.railsfromthecrypt.com/v1/notifications" \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN"

# Get unread notifications only
curl -X GET "https://api.railsfromthecrypt.com/v1/notifications?status=unread" \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN"
```

```javascript JavaScript
// Get all notifications
const response = await fetch('https://api.railsfromthecrypt.com/v1/notifications', {
  headers: {
    'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
  }
});

const data = await response.json();

// Get unread notifications
const unreadResponse = await fetch('https://api.railsfromthecrypt.com/v1/notifications?status=unread', {
  headers: {
    'Authorization': 'Bearer YOUR_ACCESS_TOKEN'
  }
});

const unreadData = await unreadResponse.json();
```

</CodeGroup>
