## Overview

Check the Know Your Business (KYB) verification status for a client. This endpoint provides real-time information about the KYB verification process, including document status and overall progress.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

<ParamField query="client_id" type="string" required>
  Client ID to check KYB status for.
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the KYB status was retrieved successfully.
</ResponseField>

<ResponseField name="data" type="object">
  KYB status data.
</ResponseField>

<ResponseField name="data.client_id" type="string">
  Client ID.
</ResponseField>

<ResponseField name="data.kyb_status" type="string">
  Overall KYB verification status.
</ResponseField>

<ResponseField name="data.verification_id" type="string">
  Unique verification ID.
</ResponseField>

<ResponseField name="data.initiated_at" type="string">
  When KYB verification was initiated.
</ResponseField>

<ResponseField name="data.completed_at" type="string">
  When KYB verification was completed (if applicable).
</ResponseField>

<ResponseField name="data.expires_at" type="string">
  When the KYB verification expires.
</ResponseField>

<ResponseField name="data.progress" type="object">
  KYB verification progress information.
</ResponseField>

<ResponseField name="data.progress.completed_documents" type="number">
  Number of completed documents.
</ResponseField>

<ResponseField name="data.progress.total_documents" type="number">
  Total number of required documents.
</ResponseField>

<ResponseField name="data.progress.percentage" type="number">
  Completion percentage.
</ResponseField>

<ResponseField name="data.documents" type="array">
  List of document statuses.
</ResponseField>

<ResponseField name="data.documents[].type" type="string">
  Document type.
</ResponseField>

<ResponseField name="data.documents[].name" type="string">
  Document name.
</ResponseField>

<ResponseField name="data.documents[].status" type="string">
  Document verification status.
</ResponseField>

<ResponseField name="data.documents[].uploaded_at" type="string">
  When the document was uploaded.
</ResponseField>

<ResponseField name="data.documents[].reviewed_at" type="string">
  When the document was reviewed.
</ResponseField>

<ResponseField name="data.documents[].reviewer_notes" type="string">
  Notes from the reviewer.
</ResponseField>

<ResponseField name="data.documents[].required" type="boolean">
  Whether the document is required.
</ResponseField>

<ResponseField name="data.verification_notes" type="string">
  Overall verification notes.
</ResponseField>

<ResponseField name="data.next_steps" type="array">
  List of next steps for the client.
</ResponseField>

## Implementation Example

<CodeGroup>

```javascript JavaScript
class KYBStatusChecker {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async checkKYBStatus(clientId) {
    try {
      const response = await this.apiClient.get(`/v1/client/kyb/status?client_id=${clientId}`);
      
      if (response.success) {
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to check KYB status:', error);
      throw error;
    }
  }
  
  displayKYBStatus(kybData) {
    const statusContainer = document.getElementById('kyb-status-container');
    
    statusContainer.innerHTML = `
      <div class="kyb-status-card">
        <h3>KYB Verification Status</h3>
        <div class="status-overview">
          <div class="status-badge ${kybData.kyb_status}">
            ${kybData.kyb_status.replace('_', ' ').toUpperCase()}
          </div>
          <div class="progress-bar">
            <div class="progress-fill" style="width: ${kybData.progress.percentage}%"></div>
          </div>
          <div class="progress-text">
            ${kybData.progress.completed_documents}/${kybData.progress.total_documents} documents completed
            (${kybData.progress.percentage}%)
          </div>
        </div>
        
        <div class="verification-details">
          <p><strong>Verification ID:</strong> ${kybData.verification_id}</p>
          <p><strong>Initiated:</strong> ${new Date(kybData.initiated_at).toLocaleString()}</p>
          ${kybData.completed_at ? `<p><strong>Completed:</strong> ${new Date(kybData.completed_at).toLocaleString()}</p>` : ''}
          <p><strong>Expires:</strong> ${new Date(kybData.expires_at).toLocaleString()}</p>
        </div>
        
        ${kybData.verification_notes ? `
          <div class="verification-notes">
            <h4>Verification Notes</h4>
            <p>${kybData.verification_notes}</p>
          </div>
        ` : ''}
        
        ${kybData.next_steps.length > 0 ? `
          <div class="next-steps">
            <h4>Next Steps</h4>
            <ul>
              ${kybData.next_steps.map(step => `<li>${step}</li>`).join('')}
            </ul>
          </div>
        ` : ''}
      </div>
    `;
  }
  
  displayDocumentStatus(documents) {
    const documentsContainer = document.getElementById('documents-status');
    
    documentsContainer.innerHTML = `
      <h3>Document Status</h3>
      <div class="documents-list">
        ${documents.map(doc => `
          <div class="document-item ${doc.status}">
            <div class="document-info">
              <h4>${doc.name}</h4>
              <div class="document-status">
                <span class="status-badge ${doc.status}">${doc.status.replace('_', ' ').toUpperCase()}</span>
                ${doc.required ? '<span class="required-badge">Required</span>' : '<span class="optional-badge">Optional</span>'}
              </div>
              ${doc.uploaded_at ? `<p><strong>Uploaded:</strong> ${new Date(doc.uploaded_at).toLocaleString()}</p>` : ''}
              ${doc.reviewed_at ? `<p><strong>Reviewed:</strong> ${new Date(doc.reviewed_at).toLocaleString()}</p>` : ''}
              ${doc.reviewer_notes ? `<p><strong>Notes:</strong> ${doc.reviewer_notes}</p>` : ''}
            </div>
          </div>
        `).join('')}
      </div>
    `;
  }
  
  async monitorKYBStatus(clientId, interval = 30000) {
    const checkStatus = async () => {
      try {
        const kybData = await this.checkKYBStatus(clientId);
        
        // Update UI
        this.displayKYBStatus(kybData);
        this.displayDocumentStatus(kybData.documents);
        
        // Check if verification is complete
        if (kybData.kyb_status === 'approved' || kybData.kyb_status === 'rejected') {
          clearInterval(this.statusInterval);
          console.log('KYB verification completed:', kybData.kyb_status);
        }
        
        return kybData;
      } catch (error) {
        console.error('Failed to check KYB status:', error);
      }
    };
    
    // Initial check
    await checkStatus();
    
    // Set up interval for monitoring
    this.statusInterval = setInterval(checkStatus, interval);
  }
  
  stopMonitoring() {
    if (this.statusInterval) {
      clearInterval(this.statusInterval);
      this.statusInterval = null;
    }
  }
}

// Usage
const kybStatusChecker = new KYBStatusChecker(apiClient);

// Check KYB status
document.getElementById('check-kyb-status').addEventListener('click', async () => {
  const clientId = 'client_1234567890';
  
  try {
    const kybData = await kybStatusChecker.checkKYBStatus(clientId);
    kybStatusChecker.displayKYBStatus(kybData);
    kybStatusChecker.displayDocumentStatus(kybData.documents);
  } catch (error) {
    console.error('Failed to check KYB status:', error);
  }
});

// Start monitoring KYB status
document.getElementById('start-monitoring').addEventListener('click', async () => {
  const clientId = 'client_1234567890';
  await kybStatusChecker.monitorKYBStatus(clientId);
});

// Stop monitoring
document.getElementById('stop-monitoring').addEventListener('click', () => {
  kybStatusChecker.stopMonitoring();
});
```
