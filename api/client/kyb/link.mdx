## Overview

Get a Know Your Business (KYB) verification link for a client. This endpoint generates a secure link that can be used to access the KYB verification process.

<Note>
  This endpoint requires authentication via Bearer token in the Authorization header.
</Note>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication. Format: `Bearer <access_token>`
</ParamField>

<ParamField query="client_id" type="string" required>
  Client ID to get KYB link for.
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the KYB link was generated successfully.
</ResponseField>

<ResponseField name="message" type="string">
  Success message.
</ResponseField>

<ResponseField name="data" type="object">
  KYB link data.
</ResponseField>

<ResponseField name="data.verification_link" type="string">
  KYB verification link URL.
</ResponseField>

<ResponseField name="data.client_id" type="string">
  Client ID.
</ResponseField>

<ResponseField name="data.expires_at" type="string">
  Link expiration timestamp.
</ResponseField>

<ResponseField name="data.verification_id" type="string">
  Unique verification ID.
</ResponseField>

<ResponseField name="data.status" type="string">
  Current verification status.
</ResponseField>

<ResponseField name="data.required_documents" type="array">
  List of required documents for KYB verification.
</ResponseField>

<ResponseField name="data.required_documents[].type" type="string">
  Document type.
</ResponseField>

<ResponseField name="data.required_documents[].name" type="string">
  Document name.
</ResponseField>

<ResponseField name="data.required_documents[].description" type="string">
  Document description.
</ResponseField>

<ResponseField name="data.required_documents[].required" type="boolean">
  Whether the document is required.
</ResponseField>

<ResponseField name="data.required_documents[].uploaded" type="boolean">
  Whether the document has been uploaded.
</ResponseField>

<ResponseField name="data.required_documents[].status" type="string">
  Document verification status.
</ResponseField>

## Implementation Example

<CodeGroup>

```javascript JavaScript
class KYBLinkManager {
  constructor(apiClient) {
    this.apiClient = apiClient;
  }
  
  async getKYBLink(clientId) {
    try {
      const response = await this.apiClient.get(`/v1/client/kyb/link?client_id=${clientId}`);
      
      if (response.success) {
        this.storeKYBLink(response.data);
        return response.data;
      } else {
        throw new Error(response.message);
      }
    } catch (error) {
      console.error('Failed to get KYB link:', error);
      throw error;
    }
  }
  
  storeKYBLink(kybData) {
    // Store KYB link data
    localStorage.setItem('kyb_link', kybData.verification_link);
    localStorage.setItem('kyb_verification_id', kybData.verification_id);
    localStorage.setItem('kyb_data', JSON.stringify(kybData));
  }
  
  getKYBLink() {
    return localStorage.getItem('kyb_link');
  }
  
  getKYBVerificationId() {
    return localStorage.getItem('kyb_verification_id');
  }
  
  getKYBData() {
    const kybData = localStorage.getItem('kyb_data');
    return kybData ? JSON.parse(kybData) : null;
  }
  
  clearKYBData() {
    localStorage.removeItem('kyb_link');
    localStorage.removeItem('kyb_verification_id');
    localStorage.removeItem('kyb_data');
  }
  
  displayKYBLink(kybData) {
    const linkContainer = document.getElementById('kyb-link-container');
    
    linkContainer.innerHTML = `
      <div class="kyb-link-card">
        <h3>KYB Verification Link</h3>
        <p>Click the link below to start the KYB verification process:</p>
        <a href="${kybData.verification_link}" 
           target="_blank" 
           class="kyb-link-button">
          Start KYB Verification
        </a>
        <div class="kyb-details">
          <p><strong>Verification ID:</strong> ${kybData.verification_id}</p>
          <p><strong>Expires:</strong> ${new Date(kybData.expires_at).toLocaleString()}</p>
          <p><strong>Status:</strong> <span class="status ${kybData.status}">${kybData.status}</span></p>
        </div>
      </div>
    `;
  }
  
  displayRequiredDocuments(requiredDocuments) {
    const documentsContainer = document.getElementById('required-documents');
    
    documentsContainer.innerHTML = `
      <h3>Required Documents</h3>
      <div class="documents-list">
        ${requiredDocuments.map(doc => `
          <div class="document-item ${doc.required ? 'required' : 'optional'}">
            <div class="document-info">
              <h4>${doc.name}</h4>
              <p>${doc.description}</p>
              <div class="document-status">
                <span class="status ${doc.status}">${doc.status}</span>
                <span class="uploaded ${doc.uploaded ? 'yes' : 'no'}">
                  ${doc.uploaded ? 'Uploaded' : 'Not Uploaded'}
                </span>
              </div>
            </div>
          </div>
        `).join('')}
      </div>
    `;
  }
  
  async initiateKYBVerification(clientId) {
    try {
      const kybData = await this.getKYBLink(clientId);
      
      // Display KYB link
      this.displayKYBLink(kybData);
      
      // Display required documents
      this.displayRequiredDocuments(kybData.required_documents);
      
      return kybData;
    } catch (error) {
      console.error('Failed to initiate KYB verification:', error);
      throw error;
    }
  }
  
  async sendKYBLinkToClient(clientId, email) {
    try {
      const kybData = await this.getKYBLink(clientId);
      
      // Send email with KYB link
      const emailData = {
        to: email,
        subject: 'KYB Verification Required',
        template: 'kyb_verification',
        data: {
          verification_link: kybData.verification_link,
          client_id: clientId,
          expires_at: kybData.expires_at
        }
      };
      
      // This would typically call an email service
      console.log('Sending KYB link email:', emailData);
      
      return kybData;
    } catch (error) {
      console.error('Failed to send KYB link:', error);
      throw error;
    }
  }
}

// Usage
const kybLinkManager = new KYBLinkManager(apiClient);

// Get KYB link for client
document.getElementById('get-kyb-link').addEventListener('click', async () => {
  const clientId = 'client_1234567890';
  
  try {
    const kybData = await kybLinkManager.initiateKYBVerification(clientId);
    console.log('KYB verification link generated:', kybData);
  } catch (error) {
    console.error('Failed to get KYB link:', error);
  }
});

// Send KYB link to client
document.getElementById('send-kyb-link').addEventListener('click', async () => {
  const clientId = 'client_1234567890';
  const clientEmail = 'contact@acme.com';
  
  try {
    const kybData = await kybLinkManager.sendKYBLinkToClient(clientId, clientEmail);
    console.log('KYB link sent to client:', kybData);
  } catch (error) {
    console.error('Failed to send KYB link:', error);
  }
});
```

## KYB Verification Process

### Step-by-Step Process

<Steps>
  <Step title="Generate Link">
    Generate KYB verification link for client
  </Step>
  <Step title="Send Link">
    Send verification link to client via email
  </Step>
  <Step title="Client Access">
    Client accesses verification link
  </Step>
  <Step title="Document Upload">
    Client uploads required documents
  </Step>
  <Step title="Review Process">
    Documents are reviewed by compliance team
  </Step>
  <Step title="Approval">
    Client is approved or additional documents requested
  </Step>
</Steps>
